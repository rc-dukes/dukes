28  public class AdaFruit implements ServoCommand {
29  
30  	private BigDecimal frequency;
31  	private I2CBus bus;
32  	private PCA9685GpioProvider provider;
33  	private GpioPinPwmOutput[] outputs;
34  	private static final Logger LOG = LoggerFactory.getLogger(AdaFruit.class);
35  
36  	/**
37  	 * default constructor
38  	 * 
39  	 * @throws Exception
40  	 */
41  	public AdaFruit() throws Exception {
42  		this(48.828, 1.0578);
43  	}
44  
45  	/**
46  	 * construct me
47  	 * 
48  	 * @throws Exception
49  	 */
50  	public AdaFruit(double pFrequency, double pFrequencyCorrectionFactor)
51  			throws UnsupportedBusNumberException, Exception {
52  		// This would theoretically lead into a resolution of 5 microseconds per step:
53  		// 4096 Steps (12 Bit)
54  		// T = 4096 * 0.000005s = 0.02048s
55  		// f = 1 / T = 48.828125
56  		frequency = new BigDecimal(pFrequency);
57  		// Correction factor: actualFreq / targetFreq
58  		// e.g. measured actual frequency is: 51.69 Hz
59  		// Calculate correction factor: 51.65 / 48.828 = 1.0578
60  		// --> To measure actual frequency set frequency without correction factor(or
61  		// set to 1)
62  		BigDecimal frequencyCorrectionFactor = new BigDecimal(pFrequencyCorrectionFactor);
63  		// Create custom PCA9685 GPIO provider
64  		bus = I2CFactory.getInstance(I2CBus.BUS_1);
65  		provider = new PCA9685GpioProvider(bus, 0x40, frequency, frequencyCorrectionFactor);
66  		// Define outputs in use for this example
67  		outputs = provisionPwmOutputs(provider);
68  		// Reset outputs
69  		provider.reset();
70  	}
71  
72  	protected GpioPinPwmOutput[] provisionPwmOutputs(final PCA9685GpioProvider gpioProvider) {
73  		GpioController gpio = GpioFactory.getInstance();
74  		GpioPinPwmOutput myOutputs[] = { gpio.provisionPwmOutputPin(gpioProvider, PCA9685Pin.PWM_00, "Pulse 00"),
75  				gpio.provisionPwmOutputPin(gpioProvider, PCA9685Pin.PWM_01, "Pulse 01"),
76  				gpio.provisionPwmOutputPin(gpioProvider, PCA9685Pin.PWM_02, "Pulse 02"),
77  				gpio.provisionPwmOutputPin(gpioProvider, PCA9685Pin.PWM_03, "Pulse 03"),
78  				gpio.provisionPwmOutputPin(gpioProvider, PCA9685Pin.PWM_04, "Pulse 04"),
79  				gpio.provisionPwmOutputPin(gpioProvider, PCA9685Pin.PWM_05, "Pulse 05"),
80  				gpio.provisionPwmOutputPin(gpioProvider, PCA9685Pin.PWM_06, "Pulse 06"),
81  				gpio.provisionPwmOutputPin(gpioProvider, PCA9685Pin.PWM_07, "Pulse 07"),
82  				gpio.provisionPwmOutputPin(gpioProvider, PCA9685Pin.PWM_08, "Pulse 08"),
83  				gpio.provisionPwmOutputPin(gpioProvider, PCA9685Pin.PWM_09, "Pulse 09"),
84  				gpio.provisionPwmOutputPin(gpioProvider, PCA9685Pin.PWM_10, "Pulse 10"),
85  				gpio.provisionPwmOutputPin(gpioProvider, PCA9685Pin.PWM_11, "Pulse 11"),
86  				gpio.provisionPwmOutputPin(gpioProvider, PCA9685Pin.PWM_12, "Pulse 12"),
87  				gpio.provisionPwmOutputPin(gpioProvider, PCA9685Pin.PWM_13, "Pulse 13"),
88  				gpio.provisionPwmOutputPin(gpioProvider, PCA9685Pin.PWM_14, "Pulse 14"),
89  				gpio.provisionPwmOutputPin(gpioProvider, PCA9685Pin.PWM_15, "Pulse 15") };
90  		return myOutputs;
91  	}
92  
93  	private static final int SERVO_DURATION_MIN = 400;
94  	private static final int SERVO_DURATION_MAX = 2400;
95  
96  	@Override
97  	public void setServo(int ioId, int value) {
98  		if ((ioId < 0) || (ioId > 15))
99  			throw new IllegalArgumentException("Invalid ioId " + ioId + " ioId must be 0-15");
100 		GpioPinPwmOutput output = outputs[ioId];
101 		int duration = SERVO_DURATION_MIN + value * (SERVO_DURATION_MAX - SERVO_DURATION_MIN) / 256;
102 		if (debug) {
103 			String msg = String.format("setting servo %3d (%s) to %4d", output.getPin().getAddress(),output.getName(), duration);
104 			LOG.info(msg);
105 		}
106 		output.setPwm(duration);
107 	}

109 	protected CmdLineParser parser;
110 	@Option(name = "-d", aliases = {
111 			"--debug" }, usage = "debug\ncreate additional debug output if this switch is used")
112 	protected boolean debug = false;
113 
114 	@Option(name = "-i", aliases = { "--ioid" }, usage = "ioid to be used")
115 	protected int ioId = 0;
116 
117 	@Option(name = "-v", aliases = { "--value" }, usage = "value to be set")
118 	protected int value = 0;
119 
120 	/**
121 	 * parse the given Arguments
122 	 * 
123 	 * @param args
124 	 * @throws CmdLineException
125 	 */
126 	public void parseArguments(String[] args) throws CmdLineException {
127 		parser = new CmdLineParser(this);
128 		parser.parseArgument(args);
129 	}
130 
131 	public void work() {
132 		this.setServo(ioId, value);
133 		try {
134 			Thread.sleep(1000);
135 		} catch (Exception ex) {
136 		}
137 	}
138 
139 	/**
140 	 * command line entry point
141 	 * 
142 	 * @param args
143 	 */
144 	public static void main(String args[]) {
145 		try {
146 			AdaFruit adaFruit = new AdaFruit();
147 			adaFruit.parseArguments(args);
148 			adaFruit.work();
149 		} catch (Exception e) {
150 			ErrorHandler.getInstance().handle(e);
151 		}
152 	}
