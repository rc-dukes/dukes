22  /**
23   * Runner to start the remote cluster on the car
24   *
25   */
26  public class RemoteCar extends DukesVerticle {
27  
28    private ClusterStarter starter;
29    private Car car;
30    private Disposable startRequester;
31    private WatchDog watchDog;
32    private CarVerticle carVerticle;
33  
34    /**
35     * construct me as a verticle
36     */
37    public RemoteCar() {
38      super(Characters.GENERAL_LEE);
39    }
40  
41    private static final Logger LOG = LoggerFactory.getLogger(RemoteCar.class);
42  
43    @Override
44    public void start() throws Exception {
45      super.preStart();
46      super.consumer(Events.START_CAR, this::startCar);
47      super.consumer(Events.STOP_CAR, this::stopCar);
48      super.consumer(Events.ECHO, this::echo);
49      // Observable<Long> startRequestRepeat = Observable.interval(5, 2,
50      // TimeUnit.SECONDS);
51      // startRequester=startRequestRepeat.subscribe(l->requestStart());
52      super.postStart();
53    }
54  
55    private void requestStart() {
56      if (car == null) {
57        super.send(Characters.BOSS_HOGG, "REQUEST_CONFIG", "true");
58      } else {
59        startRequester.dispose();
60      }
61    }
62  
63    private void echo(Message<JsonObject> message) {
64      JsonObject jo = message.body();
65      System.out.println("received jsonobject: " + jo);
66      String msg = String.format("Thread %d is clustered %s",
67          Thread.currentThread().getId(), this.getVertx().isClustered());
68      System.out.println(msg);
69      // starter.getVertx().eventBus().send(Characters.BOSS_HOGG.getCallsign(),
70      // jo);
71      super.sendEvent(Characters.ROSCO, Events.ECHO_REPLY, jo);
72      System.out.println("send echo reply to " + Characters.ROSCO + " finished");
73    }
74  
75    private void stopCar(Message<JsonObject> message) {
76      if (car == null) {
77        LOG.error("No car instance active - can't stop");
78        return;
79      }
80      car.stop();
81      starter.undeployVerticle(watchDog);
82      starter.undeployVerticle(carVerticle);
83      Car.resetInstance();
84      car = null;
85    }
86  
87    private void startCar(Message<JsonObject> message) {
88      JsonObject configJo = message.body();
89      if (configJo == null) {
90        LOG.error("Can't start verticle - configuration message body is null");
91        return;
92      }
93      Environment.from(configJo);
94      if (car != null) {
95        LOG.info(
96            "Car verticle already configured and started - only reconfigured the environment ...");
97        return;
98      }
99      LOG.info(
100         "Firing up General Lee vert.x and core controller, tutu tu tu tu tutututu tututu...");
101 
102     car = Car.getInstance(); // the one and only to be used also by the
103                              // verticle!
104     configureShutdownHook(car);
105     // Set wheel and speed to neutral.
106     car.stop();
107     // Command.stop();
108     try {
109       watchDog = new WatchDog(car);
110       carVerticle = new CarVerticle();
111       starter.deployVerticles(watchDog, carVerticle);
112     } catch (Exception e) {
113       ErrorHandler.getInstance().handle(e);
114     }
115   }
116 
117   protected CmdLineParser parser;
118   @Option(name = "-d", aliases = {
119       "--debug" }, usage = "debug\ncreate additional debug output if this switch is used")
120   protected boolean debug = false;
121   @Option(name = "-ch", aliases = { "--hostname" }, usage = "clusterHostname")
122   String clusterHostname = null;
123   @Option(name = "-ph", aliases = { "--publichost" }, usage = "public hostname")
124   String publicHost = null;
125 
126   /**
127    * parse the given Arguments
128    * 
129    * @param args
130    * @throws CmdLineException
131    */
132   public void parseArguments(String[] args) throws CmdLineException {
133     parser = new CmdLineParser(this);
134     parser.parseArgument(args);
135   }
136 
137   /**
138    * start me with given command line parameters
139    * 
140    * @param args
141    *          - the command line parameters to use
142    * @throws Exception
143    */
144   public void mainInstance(String... args) {
145     try {
146       this.parseArguments(args);
147     } catch (CmdLineException cle) {
148       ErrorHandler.getInstance().handle(cle);
149       parser.printUsage(System.err);
150       System.exit(1);
151     }
152     starter = new ClusterStarter();
153     starter.prepare();
154     starter.configureCluster(clusterHostname, publicHost);
155   
156     // bootstrap the deployment by deploying me
157     try {
158       starter.deployVerticles(this);
159     } catch (Exception e) {
160       ErrorHandler.getInstance().handle(e);
161     }
162   }
163 
164   private static void configureShutdownHook(Car car) {
165     Runtime.getRuntime().addShutdownHook(new Thread() {
166       public void run() {
167         LOG.info("Activating shutdown hook.");
168         car.stop();
169       }
170     });
171   }
172 
173   /**
174    * static main routine
175    * 
176    * @param args
177    */
178   public static void main(String... args) {
179     RemoteCar remoteCar = new RemoteCar();
180     remoteCar.mainInstance(args);
181   }
