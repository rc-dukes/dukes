3   import java.util.stream.Collectors;
4   
5   import static java.util.Arrays.stream;
6   
7   /**
8    * implementation of point with multiple dimensions
9    * with 2D and 3D interfaces
10   *
11   */
12  public class Point implements Point2D, Point3D, Comparable<Point> {
13  
14      public static final int X = 0;
15      public static final int Y = 1;
16      public static final int Z = 2;
17  
18      double[] dimensions;
19  
20      /**
21       * create a point with the given dimensions
22       * @param dimensions - the dimensions to use
23       */
24      public Point(double... dimensions) {
25          if (dimensions.length < 1) {
26              throw new IllegalArgumentException("Point must have at least 1 dimension");
27          }
28  
29          this.dimensions = dimensions;
30      }
31  
32      /**
33       * calculate the distance to another point
34       * @param dimensions - the dimensions of the other point
35       * @return - the distance
36       */
37      public double distance(double ... dimensions) {
38          if (this.dimensions.length != dimensions.length) {
39              throw new IllegalArgumentException("Can only calculate distance between two points with similar dimensions.");
40          }
41  
42          double totalDistance = 0d;
43  
44          for (int i = 0; i < this.dimensions.length; i++) {
45              double myVal = this.dimensions[i];
46              double otherVal = dimensions[i];
47  
48              totalDistance += Math.pow(myVal - otherVal, 2d);
49          }
50  
51          return Math.sqrt(totalDistance);
52      }
53  
54      @Override
55      public double distance(Point2D other) {
56          double[] otherDimensons = new double[2];
57          otherDimensons[X] = other.getX();
58          otherDimensons[Y] = other.getY();
59          return distance(otherDimensons);
60      }
61  
62      @Override
63      public double distance(Point3D other) {
64          double[] otherDimensons = new double[3];
65          otherDimensons[X] = other.getX();
66          otherDimensons[Y] = other.getY();
67          otherDimensons[Z] = other.getZ();
68          return distance(otherDimensons);
69      }
70  
71      @Override
72      public String toString() {
73          return "{" + stream(dimensions).mapToObj(String::valueOf).collect(Collectors.joining(",")) + "}";
74      }
75  
76      @Override
77      public double getX() {
78          return dimensions[X];
79      }
80  
81      @Override
82      public double getY() {
83          return dimensions[Y];
84      }
85  
86      @Override
87      public double getZ() {
88          return dimensions[Z];
89      }
90      
91      private Double getMax() {
92       Double max=Double.MIN_VALUE;
93       for (double val:dimensions) {
94         if (val>max)
95           max=val;
96       }
97       return max;
98      }
99  
100     @Override
101     public int compareTo(Point o) {
102       Double max=getMax();
103       Double omax=o.getMax();
104       return max.compareTo(omax);
105     }
106 
107 
108 }
