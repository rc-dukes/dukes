27    /**
28     * construct me from a hough lines detect result
29     * @param probabilistic
30     * @param h 
31     * @param w 
32     * @param coordinates - the 4 coordinates to construct me from
33     * @return the Line
34     */
35    public static Line fromHough(boolean probabilistic,int w, int h, double ...coordinates) {
36      if (probabilistic) {
37        return new Line(coordinates);
38      } else {
39        double rho=coordinates[0];
40        double theta= coordinates[1];
41        double a = Math.cos(theta), b = Math.sin(theta);
42        double x0 = a*rho, y0 = b*rho;
43        Point pt1 = new Point(Math.round(x0 + w*(-b)), Math.round(y0 + h*(a)));
44        Point pt2 = new Point(Math.round(x0 - w*(-b)), Math.round(y0 - h*(a)));
45        return new Line(pt1,pt2);
46      }
47    }

155   /**
156    * calculate the average line for a collection of given lines
157    * 
158    * @param lines
159    *          - the lines to calculate the average for
160    * @return the average line
161    */
162   public static Line average(Collection<Line> lines) {
163     // double sumAngleRad = 0;
164     double sumx = 0;
165     double sumy = 0;
166     double sumxlen = 0;
167     double sumylen = 0;
168 
169     for (Line line : lines) {
170       // sumAngleRad += line.angleRad();
171       sumx += line.point1.getX() + line.point2.getX();
172       sumy += line.point1.getY() + line.point2.getY();
173       sumxlen += line.point2.getX() - line.point1.getX();
174       sumylen += line.point2.getY() - line.point1.getY();
175     }
176 
177     // double avgAngleRad = sumAngleRad / lines.size();
178     double avgx = sumx / (lines.size() * 2);
179     double avgy = sumy / (lines.size() * 2);
180     double avgxlen = sumxlen / lines.size();
181     double avgylen = sumylen / lines.size();
182 
183     Point origin = new Point(avgx - 0.5 * avgxlen, avgy - 0.5 * avgylen);
184     Point2D endV = new Vector(avgxlen, avgylen).calculate(origin, 1);
185 
186     Line avgLine = new Line(origin, new Point(endV.getX(),endV.getY()));
187 
188     // System.out.println("avg rad: " + avgAngleRad + " line rad: " +
189     // avgLine.angleRad());
190 
191     return avgLine;
192   }

194   /**
195    * Tests if a point is Left|On|Right of this line it it were infinite. (See:
196    * Algorithm 1 "Area of Triangles and Polygons")
197    * 
198    * @param point
199    *          The point to determine the position relative to this line.
200    * @return >0 for P2 left of this line =0 for P2 on this line <0 for P2 right
201    *         of this line
202    */
203   private int howLeft(Point2D point) {
204     Point2D start = point1;
205     Point2D end = point2;
206     return (int) ((end.getX() - start.getX()) * (point.getY() - start.getY())
207         - (point.getX() - start.getX()) * (end.getY() - start.getY()));
208   }

239   public boolean existsOnLine(Point2D point) {
240     if (point.getY() > max(point1.getY(), point2.getY())) {
241       // Point above line
242       return false;
243     }
244 
245     if (point.getY() < min(point1.getY(), point2.getY())) {
246       // Point below line
247       return false;
248     }
249 
250     return howLeft(point) == 0;
251   }

347   /**
348    * calculate intersection to the given other line
349    * 
350    * @param other
351    *          - the other line to intersect with
352    * @return the intersection point
353    */
354   public Optional<Point2D> intersect(Line other) {
355     double x1 = point1.getX();
356     double x2 = point2.getX();
357     double x3 = other.point1.getX();
358     double x4 = other.point2.getX();
359 
360     double y1 = point1.getY();
361     double y2 = point2.getY();
362     double y3 = other.point1.getY();
363     double y4 = other.point2.getY();
364 
365     double determinant = (x1 - x2) * (y3 - y4) - (y1 - y2) * (x3 - x4);
366 
367     if (determinant == 0) {
368       // edge case: lines are parallel, there is no intersection
369       return Optional.empty();
370     }
371 
372     double x = ((x1 * y2 - y1 * x2) * (x3 - x4)
373         - (x1 - x2) * (x3 * y4 - y3 * x4)) / determinant;
374     double y = ((x1 * y2 - y1 * x2) * (y3 - y4)
375         - (y1 - y2) * (x3 * y4 - y3 * x4)) / determinant;
376 
377     return of(new Point(x, y));
378   }

380   /**
381    * calculate the distance to the given point
382    * 
383    * @param point
384    *          - the point to calculate the distance to
385    * @return the distance
386    */
387   public double distance(Point2D point) {
388     double x0 = point.getX();
389     double y0 = point.getY();
390     double x1 = point1.getX();
391     double y1 = point1.getY();
392     double x2 = point2.getX();
393     double y2 = point2.getY();
394 
395     double determinant = sqrt(pow(y2 - y1, 2) + pow(x2 - x1, 2));
396 
397     return abs((y2 - y1) * x0 - (x2 - x1) * y0 + x2 * y1 - y2 * x1)
398         / determinant;
399   }

404   public static class Vector {

409     /**
410      * create a vector from the given values
411      * 
412      * @param x
413      *          - x component of vector
414      * @param y
415      *          - y component of vector
416      */
417     public Vector(double x, double y) {
418       this.x = x;
419       this.y = y;
420     }
421 
422     public Point2D calculate(Point2D origin, double delta) {
423       return new Point(origin.getX() + delta * x, origin.getY() + delta * y);
424     }
425 
426     public double calculateX(double deltaY) {
427       return (deltaY / y) * x;
428     }
429 
430     public double calculateY(double deltaX) {
431       return (deltaX / x) * y;
432     }
433 
434     public static Vector forY(double y) {
435       return new Vector(0, y);
436     }
437 
438     public static Vector forX(double x) {
439       return new Vector(x, 0);
440     }

442     /**
443      * convert me to a String

445     public String toString() {
446       return String.format(Locale.ENGLISH, "{%f,%f}", x, y);
