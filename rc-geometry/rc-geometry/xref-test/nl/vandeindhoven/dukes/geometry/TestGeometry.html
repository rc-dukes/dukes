29    @Test
30    public void testLine() {
31      Line line = new Line(0., 0., 100., 100.);
32      Line liner = new Line(100., 100., 0.0, 0.0);
33      Line line2 = new Line(0., 10., 100, 110.);
34      Line line3 = new Line(50.0, 0, 50.0, 100.);
35      assertEquals(141.0, line.length(), 0.5);
36      assertEquals(Math.PI / 4, line.angleRad(), 0.01);
37      assertEquals(45.0, line.angleDeg(), 0.01);
38      assertEquals(0., line.getPoint1().getX(), 0.01);
39      assertEquals(0., line.getPoint1().getY(), 0.01);
40      assertEquals(100., line.getPoint2().getX(), 0.01);
41      assertEquals(100., line.getPoint2().getY(), 0.01);
42      Line avgLine = Line.average(line, line2);
43      assertEquals("{0.0,5.0} - {100.0,105.0}", avgLine.toString());
44      Point2D pRight = new Point(50.0, 40.0);
45      Point2D pLeft = new Point(50.0, 60.0);
46      Point2D pOn = new Point(50.0, 50.0);
47      assertTrue(line.isLeftOfLine(pLeft));
48      assertFalse(line.isRightOfLine(pLeft));
49      assertFalse(line.existsOnLine(pLeft));
50      assertFalse(line.isLeftOfLine(pRight));
51      assertTrue(line.isRightOfLine(pRight));
52      assertFalse(line.existsOnLine(pRight));
53      assertFalse(line.isLeftOfLine(pOn));
54      assertFalse(line.isRightOfLine(pOn));
55      assertTrue(line.existsOnLine(pOn));
56      // coverage of existsOnLine
57      assertFalse(line.existsOnLine(new Point(50.0, 120.0)));
58      assertFalse(line.existsOnLine(new Point(50.0, -20.0)));
59  
60      assertEquals("{100.0,100.0}", line.bottomMost().toString());
61      assertEquals("{100.0,100.0}", line.rightMost().toString());
62      assertEquals("{0.0,0.0}", line.topMost().toString());
63      assertEquals("{0.0,0.0}", line.leftMost().toString());
64      assertEquals("{100.0,100.0}", liner.bottomMost().toString());
65      assertEquals("{100.0,100.0}", liner.rightMost().toString());
66      assertEquals("{0.0,0.0}", liner.topMost().toString());
67      assertEquals("{0.0,0.0}", liner.leftMost().toString());
68      assertEquals("{50.0,50.0}", line.pointAt(0.5).toString());
69      assertEquals("{50.0,50.0}", line.intersect(line3).get().toString());
70      assertFalse(line.intersect(line2).isPresent());
71      assertEquals(7.071, line.distance(pLeft), 0.0001);
72    }

83    @Test
84    public void testPoint() {
85      try {
86        new Point();
87        fail("exception expected");
88      } catch (IllegalArgumentException e) {
89        assertEquals("Point must have at least 1 dimension", e.getMessage());
90      }
91      Point p = new Point(0, 0);
92      try {
93        p.distance();
94        fail("exception expected");
95      } catch (IllegalArgumentException e) {
96        assertEquals(
97            "Can only calculate distance between two points with similar dimensions.",
98            e.getMessage());
99      }
100     Point3D p3 = new Point(0, 0, 0);
101     assertEquals(0.0, p3.getZ(), 0.001);
102     assertEquals(Math.sqrt(3), p3.distance(new Point(1, 1, 1)), 0.00001);
103   }

105   @Test
106   public void testPolygon() {
107     Polygon polygons[] = { Polygon.square(new Point(0, 0), new Point(100, 100)),
108         new Polygon(new Point(0, 0), new Point(100, 100), new Point(100, 0),
109             new Point(0, 100)) };
110     double expectedShoelace[] = { 20000, 0 };
111     String expectedIntersects[] = { "[{0.0,25.0}, {100.0,25.0}]",
112         "[{25.0,25.0}, {75.0,25.0}, {100.0,25.0}]" };
113     int index = 0;
114     Line line = new Line(0, 25, 100, 25);
115     for (Polygon polygon : polygons) {
116       List<Point2D> points = polygon.getPoints();
117       List<Line> edges = polygon.edges();
118       List<Point2D> cpoints = polygon.getPointsClockwise();
119       List<Point2D> ccpoints = polygon.getPointsCounterClockwise();
120       assertEquals(4, points.size());
121       assertEquals(4, edges.size());
122       assertEquals(4, cpoints.size());
123       assertEquals(4, ccpoints.size());
124       for (int i = 0; i < points.size(); i++) {
125         assertEquals(points.get(i), cpoints.get(i));
126         assertEquals(points.get(i), ccpoints.get(3 - i));
127       }
128       assertEquals(expectedShoelace[index], polygon.shoelace(), 0.01);
129       assertEquals(expectedIntersects[index],
130           polygon.intersect(line).toString());
131       index++;
132     }
133     Point point = new Point(0.0, 12.5);
134     PointInPlane pips[] = { new WindingNumbers(), 
135         //new CrossingNumber() 
136         };
137     for (PointInPlane pip : pips) {
138       assertTrue(pip.isPointInPlane(point, polygons[0]));
139     }
