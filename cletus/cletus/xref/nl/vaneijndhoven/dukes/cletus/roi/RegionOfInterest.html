1   package nl.vaneijndhoven.dukes.cletus.roi;

3   import org.opencv.core.Mat;
4   import org.opencv.core.Point;
5   import org.opencv.core.Rect;
6   import org.opencv.core.Size;
7   
8   public class RegionOfInterest {
9   
10      private final double xFraction;
11      private final double yFraction;
12      private final double widthFraction;
13      private final double heightFraction;
14  
15      public RegionOfInterest(double xFraction, double yFraction, double widthFraction, double heightFraction) {
16          this.xFraction = xFraction;
17          this.yFraction = yFraction;
18          this.widthFraction = widthFraction;
19          this.heightFraction = heightFraction;
20      }
21  
22      public Rect roi(Size base) {
23          double roiX = base.height * xFraction;
24          double roiY = base.height * yFraction;
25  
26          double roiHeight = base.height * heightFraction;
27          double roiWidth = base.width * widthFraction;
28  
29          Point origin = new Point(roiX, roiY);
30          Size roiSize = new Size(roiWidth, roiHeight);
31  
32          return new Rect(origin, roiSize);
33      }
34  
35      public Mat region(Mat image) {
36          // Use 2-arg constructor to create region image backed by original image data
37          return new Mat(image, roi(image.size()));
38      }
39  
40  }
