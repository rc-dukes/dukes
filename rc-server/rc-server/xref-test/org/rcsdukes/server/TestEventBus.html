37    /**
38     * check the start of the clusteredVertex with the given options
39     * 
40     * @param options
41     * @param timeOutMSecs
42     * @throws InterruptedException
43     */
44    public void checkStart(VertxOptions options, int timeOutMSecs)
45        throws InterruptedException {
46      boolean ready[] = { false };
47      Vertx.clusteredVertx(options, res -> {
48        if (res.succeeded()) {
49          Vertx vertx = res.result();
50          EventBus eventBus = vertx.eventBus();
51          System.out.println("We now have a clustered event bus: " + eventBus);
52          ready[0] = true;
53        } else {
54          System.out.println("Failed: " + res.cause());
55        }
56      });
57      int msecs = 0;
58      // needs some 4-8 secs on a laptop
59      // allow 10x as much for travis
60      while (!ready[0] && msecs++ < timeOutMSecs)
61        Thread.sleep(1);
62      System.out.println(String.format("%s startup after %d msecs",
63          ready[0] ? "finished" : "timed out", msecs));
64  
65    }

67    /**
68     * see https://stackoverflow.com/a/13817240/1497139
69     * 
70     * @param path
71     *          - where to create the keystore
72     * @throws Exception 
73     */
74    public void createKeyStore(String path, String pPassword) throws Exception {
75  
76      KeyStore ks = KeyStore.getInstance(KeyStore.getDefaultType());
77  
78      char[] password = pPassword.toCharArray();
79      ks.load(null, null);
80  
81      // Store away the keystore.
82      FileOutputStream fos = new FileOutputStream(path);
83      ks.store(fos, password);
84      fos.close();
85    }

87    @Test
88    public void testExample13() throws Exception {
89      String path="/tmp/keystoreExample13.jks";
90      String password="wibble";
91      this.createKeyStore(path,password);
92      VertxOptions options = new VertxOptions()
93          .setEventBusOptions(new EventBusOptions().setSsl(true)
94              .setKeyStoreOptions(
95                  new JksOptions().setPath(path).setPassword(password))
96              .setTrustStoreOptions(
97                  new JksOptions().setPath(path).setPassword(password))
98              .setClientAuth(ClientAuth.REQUIRED));
99      this.checkStart(options, 80000);
100   }
