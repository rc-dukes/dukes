22  public class TestAction {
23    public static boolean debug = true;
24    public static int TIME_OUT = 500; // we expect a result within 50 msecs
25  
26    /**
27     * check the observed value https://stackoverflow.com/a/31330706/1497139
28     * 
29     * @param obs
30     *          - the observable to check
31     * @param timeOutMilliSecs
32     *          - the number of milliseconds until the observable will timeout
33     * @param nameValues
34     *          - an array of name - value object pairs
35     */
36    public void check(Observable<JsonObject> obs, int timeOutMilliSecs,
37        String... nameValues) {
38      if (nameValues.length % 2 != 0)
39        throw new IllegalArgumentException(
40            "nameValue parameter list length may not be odd");
41      boolean observed[] = { false };
42      obs.timeout(timeOutMilliSecs, TimeUnit.MILLISECONDS).subscribe(nav -> {
43        observed[0] = true;
44        if (debug)
45          for (String key : nav.fieldNames()) {
46            System.out.println(String.format("%s=%s", key, nav.getValue(key)));
47          }
48        for (int i = 0; i < nameValues.length; i += 2) {
49          String name = nameValues[i].toString();
50          String value = nameValues[i + 1].toString();
51          String foundValue = nav.getValue(name).toString();
52          assertEquals(value, foundValue);
53        }
54      });
55      String msg = String.format(
56          "the observable wasn't observed (yet ...) after %d mSecs",
57          timeOutMilliSecs);
58      assertTrue(msg, observed[0]);
59    }
60  
61    @Test
62    public void testStraightLaneNavigator() {
63      if (!TestSuite.isTravis()) {
64        StraightLaneNavigator navigator = new StraightLaneNavigator();
65        JsonObject laneDetectResult = new JsonObject();
66        Observable<JsonObject> nav = navigator.navigate(laneDetectResult);
67        // we expect an empty Observable here
68        assertNotNull(nav);
69        // since we get no angle found for 1000ms, emergency stop
70        assertNotEquals(Observable.empty(), nav);
71        laneDetectResult.put("angle", 10.0);
72        Observable<JsonObject> nav2 = navigator.navigate(laneDetectResult);
73        check(nav2, TIME_OUT, "type", "servoDirect", "position", "60.0");
74      }
75    }
