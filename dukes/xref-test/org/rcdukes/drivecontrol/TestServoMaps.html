81    private void checkServoRangeMap(ServoRangeMap map, int[] percents,
82        int[] expectedServoPos, double[] expectedValues) {
83      ServoPosition cpos = map.getCurrentPosition();
84      assertEquals(0, cpos.getValue(), 0.0001);
85      int index = 0;
86      for (int percent : percents) {
87        map.newPosition(map.atPercent(percent));
88        cpos = map.getCurrentPosition();
89        if (debug)
90          System.out.println(map.positionInfo());
91        String hint=String.format("expected %.1f%s (%3d) at index %3d",expectedValues[index],map.getUnit(),expectedServoPos[index],index);
92        assertEquals(hint,expectedServoPos[index], cpos.getServoPos());
93        assertEquals(hint,expectedValues[index], cpos.getValue(), 0.05);
94        index++;
95      }
96      map.newPosition(map.getRange().getMin());
97      int maxPos=map.getRange().getMax().getServoPos();
98      while (map.getCurrentPosition().getServoPos() < maxPos) {
99        if (debug)
100         System.out.println(map.positionInfo());
101       cpos = map.getCurrentPosition();
102       String hint=String.format("expected %.1f%s (%3d) at index %3d",expectedValues[index],map.getUnit(),expectedServoPos[index],index);
103       assertEquals(hint,expectedServoPos[index], cpos.getServoPos());
104       assertEquals(hint,expectedValues[index], cpos.getValue(), 0.05);
105       index++;
106       map.step(1);
107     }
108   }

110   public void check(int servoPos,ServoRange range,ServoSide side, boolean expected) {
111     int clampedServoPos=range.clampServoPos(servoPos);
112     boolean in=side.isServoPosOnSide(clampedServoPos);
113     String msg=String.format("servopos %d clamped to %d should be on side %s within %d and %d",servoPos,clampedServoPos,side.getName(),side.getMin().getServoPos(),side.getMax().getServoPos());
114     assertTrue(msg,in==expected);
115   }

117   @Test
118   public void testIsOnSide() {
119     ServoPosition minLeft=new ServoPosition(130,-20);
120     ServoPosition maxLeft=new ServoPosition(100,-10);
121     ServoSide left=new ServoSide("left", -1.0, minLeft,maxLeft);
122     ServoPosition minRight=new ServoPosition(140,10);
123     ServoPosition maxRight=new ServoPosition(160,20);
124     ServoSide right=new ServoSide("right", 1.0, minRight,maxRight);
125     ServoPosition center=new ServoPosition(135,0);
126     ServoRange range=new ServoRange(1, left,center, right);
127     
128     check(115,range,left,true);
129     check(90,range,left,true);
130     check(135,range,left,false);
131     check(150,range,right,true);
132     check(180,range,right,true);
133     check(135,range,right,false);
134   }
