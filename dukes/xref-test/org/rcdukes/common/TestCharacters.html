17    public String spc(int numberOfSpaces) {
18      return String.format(String.format("%%%ds", numberOfSpaces), "");
19    }
20    
21    public String capitalize(String name) {
22      // return name.substring(0,1).toUpperCase() + name.substring(1).toLowerCase();
23      return WordUtils.capitalizeFully(name.replace("_"," ")).replace(" ","");
24    }
25    
26    public void output(String format,Object ...args) {
27      System.out.println(String.format(format, args));
28    }
29    @Test
30    public void testWikiMarkup() {
31      Characters[] chars = Characters.values();
32      Arrays.sort(chars, Comparator.comparing(Characters::getModule));
33  
34      for (Characters c : chars) {
35        String name=c.module.replace("-","");
36        String uml="https://rc-dukes.github.io/dukes/dukes/apidocs/nl/vaneijndhoven/dukes/%s/package-summary.html";
37        output("  %-14s[",name);
38        output("  %s  label=\" \"",spc(14));
39        output("  %s  image=\"%s\"",spc(14),capitalize(c.name())+".png");
40        output("  %s  fontcolor=white",spc(14));
41        output("  %s  URL=\"%s\"",spc(14),c.getUrl());
42        output("  %s]",spc(14));
43        output("  %-14s[",name+"label");
44        output("  %s  shape=\"note\"",spc(14));
45        output("  %s  label=\"%s\\n%s\"",spc(14),c.module,c.name());
46        output("  %s  URL=\""+uml+"\"",spc(14),c.module);
47        output("  %s]",spc(14));
48        output("  dukes--%slabel--%s",name,name);
49      }
50    }
