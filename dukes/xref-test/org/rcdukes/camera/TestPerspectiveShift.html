20    @Test
21    public void testChessboard() {
22      CameraMatrix matrix = new CameraMatrix(7, 7);
23      Mat chessboard = Imgcodecs.imread(testPath + "dukes_chessBoard008.png");
24      Point[] corners = matrix.findOuterChessBoardCornerPoints(chessboard);
25      assertNotNull(corners);
26      assertEquals(4, corners.length);
27      if (debug) {
28        System.out.print("corners: ");
29        String delim = "";
30        for (int i = 0; i < 4; i++) {
31          System.out.print(
32              String.format("(%.1f,%.1f)%s", corners[i].x, corners[i].y, delim));
33          delim = ",";
34          if (i == 2)
35            delim = "\n";
36        }
37      }
38      writeImage("chessBoardCorners", chessboard);
39      Size size = chessboard.size();
40      Polygon imagePolygon = new ImagePolygon(size, 0, 0, 1, 0, 1, 1, 0, 1);
41      Polygon worldPolygon = new ImagePolygon(corners);
42      PerspectiveShift perspectiveShift = new PerspectiveShift(imagePolygon,
43          worldPolygon);
44      if (debug) {
45        System.out.println("image: " + perspectiveShift.imagePoints);
46        System.out.println("world: " + perspectiveShift.worldPoints);
47      }
48      Mat shifted = perspectiveShift.apply(chessboard);
49      writeImage("shiftedChessBoard", shifted);
50    }
