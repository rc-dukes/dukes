101   @Test
102   public void testROI() throws Exception {
103     if (!isTravis()) {
104       NativeLibrary.load();
105       File imgRoot = new File(testPath);
106       assertTrue(imgRoot.isDirectory());
107       Mat image = Imgcodecs.imread(testPath + "dukes_roi_test_image.jpg");
108       assertNotNull(image);
109       assertEquals(960, image.height());
110       assertEquals(1280, image.width());
111       ROI rois[] = { new ROI("all", 0, 0, 1, 1), new ROI("far", 0, 0, 1, 0.5),
112           new ROI("near", 0, 0.5, 1, 0.5), new ROI("left", 0, 0, 0.5, 1),
113           new ROI("right", 0.5, 0, 0.5, 1) };
114       for (ROI roi : rois) {
115         System.out.println(roi);
116         Mat roiImage = roi.region(image);
117         if (debug)
118           System.out.println(String.format("%10s: %4d x %4d", roi.name,
119               roiImage.width(), roiImage.height()));
120         assertEquals(roiImage.width(), image.width() * roi.rw, 0.1);
121         assertEquals(roiImage.height(), image.height() * roi.rh, 0.1);
122         this.writeImage("dukes_roi_" + roi.name, roiImage);
123       }
124     }
