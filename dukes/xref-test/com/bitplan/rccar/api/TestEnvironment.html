22    /**
23     * set some properties for test purpose
24     * 
25     * @param nameValues
26     * @return the properties file
27     * @throws IOException
28     */
29    public File setProperties(String... nameValues) throws IOException {
30      File propFile = File.createTempFile("testEnvironment", "dukes.ini");
31      String props = "# test properties\n";
32      for (int i = 0; i < nameValues.length; i += 2) {
33        props += nameValues[i] + "=" + nameValues[i + 1] + "\n";
34      }
35      FileUtils.writeStringToFile(propFile, props, "UTF-8");
36      Environment.propFilePath = propFile.getPath();
37      // make sure the properties are read again
38      Environment.reset();
39      return propFile;
40    }
41  
42    @Test
43    public void testEnvironment() throws Exception {
44      String ip = "1.2.3.4";
45      File propFile = setProperties("targetHost", ip);
46  
47      Environment env = Environment.getInstance();
48      String piIp = env.getPiAddress();
49      // check that the configured ip address is the expected one
50      assertEquals(ip, piIp);
51      // check that we are not running on a PI - we deploy the code remotely so
52      // tests are done on a development machine e.g. laptop
53      // comment out if you actually intend to test on the PI
54      assertFalse(env.isPi());
55      // cleanup
56      propFile.delete();
57    }
58  
59    @Test
60    public void testCameraUrl() throws Exception {
61      String ip = "5.6.7.8";
62      File propFile = setProperties("targetHost", ip);
63      assertEquals("http://5.6.7.8/html/cam_pic_new.php",
64          Environment.getInstance().getCameraUrl());
65      // cleanup
66      propFile.delete();
67    }
