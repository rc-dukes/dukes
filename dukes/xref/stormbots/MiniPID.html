3   * Small, easy to use PID implementation with advanced controller capability.<br> 
4   * Minimal usage:<br>
5   * MiniPID pid = new MiniPID(p,i,d); <br>
6   * ...looping code...{ <br>
7   *   output= pid.getOutput(sensorvalue,target); <br>
8   * }
9   * 
10  * @see http://brettbeauregard.com/blog/2011/04/improving-the-beginners-pid-direction/improving-the-beginners-pid-introduction
11  */

75  	/**
76  	 * Configure the Proportional gain parameter. <br>
77  	 * This responds quickly to changes in setpoint, and provides most of the initial driving force
78  	 * to make corrections. <br>
79  	 * Some systems can be used with only a P gain, and many can be operated with only PI.<br>
80  	 * For position based controllers, this is the first parameter to tune, with I second. <br>
81  	 * For rate controlled systems, this is often the second after F.
82  	 *  
83  	 * @param p Proportional gain. Affects output according to <b>output+=P*(setpoint-current_value)</b>
84  	 */
85  	public void setP(double p){
86  		P=p;
87  		checkSigns();
88  	}
89  
90  	/**
91  	 * Changes the I parameter <br>
92  	 * This is used for overcoming disturbances, and ensuring that the controller always gets to the control mode. 
93  	 * Typically tuned second for "Position" based modes, and third for "Rate" or continuous based modes. <br>
94  	 * Affects output through <b>output+=previous_errors*Igain ;previous_errors+=current_error</b>
95  	 * 
96  	 * @see {@link #setMaxIOutput(double) setMaxIOutput} for how to restrict
97  	 *
98  	 * @param i New gain value for the Integral term
99  	 */
100 	public void setI(double i){
101 		if(I!=0){
102 			errorSum=errorSum*I/i;
103 			}
104 		if(maxIOutput!=0){
105 			maxError=maxIOutput/i;
106 		}
107 		I=i;
108 		checkSigns();
109 		 // Implementation note: 
110 		 // This Scales the accumulated error to avoid output errors. 
111 		 // As an example doubling the I term cuts the accumulated error in half, which results in the 
112 		 // output change due to the I term constant during the transition. 
113 	}
114 
115 	/**
116 	 * Changes the D parameter <br>
117 	 * This has two primary effects:
118 	 * <list>
119 	 * <li> Adds a "startup kick" and speeds up system response during setpoint changes
120 	 * <li> Adds "drag" and slows the system when moving toward the target
121 	 * </list>
122 	 * A small D value can be useful for both improving response times, and preventing overshoot.
123 	 * However, in many systems a large D value will cause significant instability, particularly 
124 	 * for large setpoint changes.
125 	 * <br>
126 	 * Affects output through <b>output += -D*(current_input_value - last_input_value)</b>
127 	 *
128 	 * @param d New gain value for the Derivative term
129 	 */
130 	public void setD(double d){
131 		D=d;
132 		checkSigns();
133 		}
134

136 	 * Configure the FeedForward parameter. <br>
137 	 * This is excellent for velocity, rate, and other  continuous control modes where you can 
138 	 * expect a rough output value based solely on the setpoint.<br>
139 	 * Should not be used in "position" based control modes.<br>
140 	 * Affects output according to <b>output+=F*Setpoint</b>. Note, that a F-only system is actually open loop.
141 	 * 
142 	 * @param f Feed forward gain. 
143 	 */
144 	public void setF(double f){
145 		F=f;
146 		checkSigns();
147 	}

180 	/**
181 	 * Set the maximum output value contributed by the I component of the system
182 	 * This can be used to prevent large windup issues and make tuning simpler
183 	 * @param maximum. Units are the same as the expected output value
184 	 */
185 	public void setMaxIOutput(double maximum){
186 		// Internally maxError and Izone are similar, but scaled for different purposes. 
187 		// The maxError is generated for simplifying math, since calculations against 
188 		// the max error are far more common than changing the I term or Izone. 
189 		maxIOutput=maximum;
190 		if(I!=0){
191 			maxError=maxIOutput/I;
192 		}
193 	}

205 	/**
206 	 * Specify a  maximum output.
207 	 * When two inputs specified, output range is configured to 
208 	 * <b>[minimum, maximum]</b>
209 	 * @param minimum possible output value
210 	 * @param maximum possible output value
211 	 */
212 	public void setOutputLimits(double minimum,double maximum){
213 		if(maximum<minimum)return;
214 		maxOutput=maximum;
215 		minOutput=minimum;
216 
217 		// Ensure the bounds of the I term are within the bounds of the allowable output swing
218 		if(maxIOutput==0 || maxIOutput>(maximum-minimum) ){
219 			setMaxIOutput(maximum-minimum);
220 		}
221 	}

246 	/**
247 	 * Calculate the output value for the current PID cycle.<br>
248 	 * @param actual The monitored value, typically as a sensor input.
249 	 * @param setpoint The target value for the system
250 	 * @return calculated output value for driving the system
251 	 */
252 	public double getOutput(double actual, double setpoint){
253 		double output;
254 		double Poutput;
255 		double Ioutput;
256 		double Doutput;
257 		double Foutput;
258 
259 		this.setpoint=setpoint;
260 
261 		// Ramp the setpoint used for calculations if user has opted to do so
262 		if(setpointRange!=0){
263 			setpoint=constrain(setpoint,actual-setpointRange,actual+setpointRange);
264 		}
265 
266 		// Do the simple parts of the calculations
267 		double error=setpoint-actual;
268 
269 		// Calculate F output. Notice, this depends only on the setpoint, and not the error. 
270 		Foutput=F*setpoint;
271 
272 		// Calculate P term
273 		Poutput=P*error;   
274 
275 		// If this is our first time running this, we don't actually _have_ a previous input or output. 
276 		// For sensor, sanely assume it was exactly where it is now.
277 		// For last output, we can assume it's the current time-independent outputs. 
278 		if(firstRun){
279 			lastActual=actual;
280 			lastOutput=Poutput+Foutput;
281 			firstRun=false;
282 		}
283 
284 		// Calculate D Term
285 		// Note, this is negative. This actually "slows" the system if it's doing
286 		// the correct thing, and small values helps prevent output spikes and overshoot 
287 		Doutput= -D*(actual-lastActual);
288 		lastActual=actual;
289 
290 		// The Iterm is more complex. There's several things to factor in to make it easier to deal with.
291 		// 1. maxIoutput restricts the amount of output contributed by the Iterm.
292 		// 2. prevent windup by not increasing errorSum if we're already running against our max Ioutput
293 		// 3. prevent windup by not increasing errorSum if output is output=maxOutput    
294 		Ioutput=I*errorSum;
295 		if(maxIOutput!=0){
296 			Ioutput=constrain(Ioutput,-maxIOutput,maxIOutput); 
297 		}    
298 
299 		// And, finally, we can just add the terms up
300 		output=Foutput + Poutput + Ioutput + Doutput;
301 
302 		// Figure out what we're doing with the error.
303 		if(minOutput!=maxOutput && !bounded(output, minOutput,maxOutput) ){
304 			errorSum=error; 
305 			// reset the error sum to a sane level
306 			// Setting to current error ensures a smooth transition when the P term 
307 			// decreases enough for the I term to start acting upon the controller
308 			// From that point the I term will build up as would be expected
309 		}
310 		else if(outputRampRate!=0 && !bounded(output, lastOutput-outputRampRate,lastOutput+outputRampRate) ){
311 			errorSum=error; 
312 		}
313 		else if(maxIOutput!=0){
314 			errorSum=constrain(errorSum+error,-maxError,maxError);
315 			// In addition to output limiting directly, we also want to prevent I term 
316 			// buildup, so restrict the error directly
317 		}
318 		else{
319 			errorSum+=error;
320 		}
321 
322 		// Restrict output to our specified output and ramp limits
323 		if(outputRampRate!=0){
324 			output=constrain(output, lastOutput-outputRampRate,lastOutput+outputRampRate);
325 		}
326 		if(minOutput!=maxOutput){ 
327 			output=constrain(output, minOutput,maxOutput);
328 			}
329 		if(outputFilter!=0){
330 			output=lastOutput*outputFilter+output*(1-outputFilter);
331 		}
332 
333 		// Get a test printline with lots of details about the internal 
334 		// calculations. This can be useful for debugging. 
335 		// System.out.printf("Final output %5.2f [ %5.2f, %5.2f , %5.2f  ], eSum %.2f\n",output,Poutput, Ioutput, Doutput,errorSum );
336 		// System.out.printf("%5.2f\t%5.2f\t%5.2f\t%5.2f\n",output,Poutput, Ioutput, Doutput );
337 
338 		lastOutput=output;
339 		return output;
340 	}

394 	 * <br>This limits the reactivity of P term, and restricts impact of large D term
395 	 * during large setpoint adjustments. Increases lag and I term if range is too small.

399 		setpointRange=range;

402 	/**
403      * Set a filter on the output to reduce sharp oscillations. <br>
404 	 * 0.1 is likely a sane starting value. Larger values use historical data
405 	 * more heavily, with low values weigh newer data. 0 will disable, filtering, and use 
406 	 * only the most recent value. <br>
407 	 * Increasing the filter strength will P and D oscillations, but force larger I 
408 	 * values and increase I term overshoot.<br>
409 	 * Uses an exponential wieghted rolling sum filter, according to a simple <br>
410 	 * <pre>output*(1-strength)*sum(0..n){output*strength^n}</pre> algorithm.
411 	 * @param output valid between [0..1), meaning [current output only.. historical output only)
412 	 */
413 	public void setOutputFilter(double strength){
414 		if(strength==0 || bounded(strength,0,1)){
415 		outputFilter=strength;
416 		}
417 	}

423 	/**
424 	 * Forces a value into a specific range
425 	 * @param value input value
426 	 * @param min maximum returned value
427 	 * @param max minimum value in range
428 	 * @return Value if it's within provided range, min or max otherwise 
429 	 */
430 	private double constrain(double value, double min, double max){
431 		if(value > max){ return max;}
432 		if(value < min){ return min;}
433 		return value;
434 	}  
435 
436 	/**
437 	 * Test if the value is within the min and max, inclusive
438 	 * @param value to test
439 	 * @param min Minimum value of range
440 	 * @param max Maximum value of range
441 	 * @return true if value is within range, false otherwise
442 	 */
443 	private boolean bounded(double value, double min, double max){
444 		// Note, this is an inclusive range. This is so tests like
445 		// `bounded(constrain(0,0,1),0,1)` will return false.
446 		// This is more helpful for determining edge-case behaviour
447 		// than <= is.
448 		return (min<value) && (value<max);
449 	}

451 	/**
452 	 * To operate correctly, all PID parameters require the same sign
453 	 * This should align with the {@literal}reversed value
454 	 */
455 	private void checkSigns(){
456 		if(reversed){  // all values should be below zero
457 			if(P>0) P*=-1;
458 			if(I>0) I*=-1;
459 			if(D>0) D*=-1;
460 			if(F>0) F*=-1;
461 		}
462 		else{  // all values should be above zero
463 			if(P<0) P*=-1;
464 			if(I<0) I*=-1;
465 			if(D<0) D*=-1;
466 			if(F<0) F*=-1;
467 		}
468 	}
