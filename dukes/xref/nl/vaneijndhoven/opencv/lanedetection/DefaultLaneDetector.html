1   package nl.vaneijndhoven.opencv.lanedetection;

3   import nl.vaneijndhoven.objects.ViewPort;
4   import nl.vaneijndhoven.dukes.geometry.Line;
5   import nl.vaneijndhoven.dukes.geometry.Point;
6   import nl.vaneijndhoven.dukes.geometry.Point2D;
7   import nl.vaneijndhoven.objects.Lane;
8   import nl.vaneijndhoven.objects.lane.LaneLeftBoundary;
9   import nl.vaneijndhoven.objects.lane.LaneRightBoundary;
10  
11  import java.util.Collection;
12  import java.util.Optional;
13  
14  public class DefaultLaneDetector implements LaneDetector {
15  
16      @Override
17      public Lane detect(Collection<Line> lines, ViewPort viewPort) {
18          Optional<Line> leftLine = new LaneLeftBoundary().boundary(lines);
19          Optional<Line> rightLine = new LaneRightBoundary().boundary(lines);
20  
21          final Line bottom = new Line(new Point(0d, viewPort.getHeight()), new Point(viewPort.getWidth(), viewPort.getHeight()));
22          final Line horizon = new Line(new Point(0d, 0d), new Point((double) viewPort.getWidth(), 0d));
23  
24          Optional<Line> leftBoundary = leftLine.map(line -> stretch(line, bottom, horizon));
25          Optional<Line> rightBoundary = rightLine.map(line -> stretch(line, bottom, horizon));
26  
27          Lane lane = new Lane(leftBoundary, rightBoundary);
28  
29          return lane;
30      }
31  
32      private Line stretch(Line line, Line bottom, Line horizon) {
33          Point2D lineTop = horizon.intersect(line).get();
34          Point2D lineBottom = bottom.intersect(line).get();
35          return new Line(lineBottom, lineTop);
36      }
37  }
