14  public class LineFilter {
15  
16      private final double angle;
17      private final double margin;
18      private boolean directional;
19  
20      public LineFilter(double angle, double margin, boolean directional) {
21          this.angle = angle;
22          this.margin = margin;
23          this.directional = directional;
24      }
25  
26      public Collection<Line> filter(Mat lines) {
27          Set<Line> lineObjects = new HashSet<>();
28  
29          for (int x = 0; x < lines.rows(); x++) {
30              Line line = new Line(lines.get(x, 0));
31              lineObjects.add(line);
32          }
33  
34          return filter(lineObjects);
35      }
36  
37      public Collection<Line> filter(Collection<Line> lines) {
38          double minDeg = angle - margin;
39          double maxDeg = angle + margin;
40          double minRad = toRadians(minDeg);
41          double maxRad = toRadians(maxDeg);
42  
43  //        System.out.println("min degrees: " + minDeg);
44  //        System.out.println("max degrees: " + maxDeg);
45  //        System.out.println("min radians: " + minRad);
46  //        System.out.println("max radians: " + maxRad);
47  
48          Set<Line> filtered = new HashSet<>();
49  
50          for (Line line : lines) {
51              double radian = line.angleRad();
52  
53              if (radian >= minRad && radian <= maxRad) {
54                  filtered.add(line);
55              }
56          }
57  
58          return filtered;
59      }
60  
61      public void debug(Mat lines) {
62          System.out.println("debug lines");
63          for (int x = 0; x < lines.rows(); x++) {
64              double[] data = lines.get(x, 0);
65              double x1 = data[0];
66              double y1 = data[1];
67              double x2 = data[2];
68              double y2 = data[3];
69              Point start = new Point(x1, y1);
70              Point end = new Point(x2, y2);
71  
72              double radian = atan2(end.y - start.y, end.x - start.x);
73  
74              double deg = toDegrees(radian);
75  
76              System.out.println("LINE P("+x1+","+y1+") - P("+x2+","+y2+"), radian: " + radian + ", deg: " +deg);
77          }
78      }
