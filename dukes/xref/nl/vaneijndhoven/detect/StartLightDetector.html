46      public StartLight detect(Mat image) {
47          // init
48  //        Mat blurredImage = new Mat();
49          Mat hsvImage = new Mat();
50          Mat mask = new Mat();
51          Mat morphOutput = new Mat();
52  // 768 * 576
53          // int x, int y, int width, int height
54          Rect rect = new Rect(250, 60, 40, 50);
55          frame = new Mat(image, rect);
56  //        frame = new RegionOfInterest(0.33, 0.10, 0.05, 0.05).region(image);
57  
58          // convert the frame to HSV
59          Imgproc.cvtColor(frame, hsvImage, Imgproc.COLOR_BGR2HSV);
60  
61          // get thresholding values from the UI
62          // remember: H ranges 0-180, S and V range 0-255
63          Scalar minValues = new Scalar(config.getHueStart(), config.getSaturationStart(),
64                  config.getValueStart());
65          Scalar maxValues = new Scalar(config.getHueStop(), config.getSaturationStop(),
66                  config.getValueStop());
67  
68          // show the current selected HSV range
69          String valuesToPrint = "Hue range: " + minValues.val[0] + "-" + maxValues.val[0]
70                  + "\tSaturation range: " + minValues.val[1] + "-" + maxValues.val[1] + "\tValue range: "
71                  + minValues.val[2] + "-" + maxValues.val[2];
72  //        this.onFXThread(this.hsvValuesProp, valuesToPrint);
73  
74          // threshold HSV image to select tennis balls
75          Core.inRange(hsvImage, minValues, maxValues, mask);
76          // show the partial output
77          collector.ifPresent(coll -> coll.mask(mask));
78  
79          // morphological operators
80          // dilate with large element, erode with small ones
81          Mat dilateElement = Imgproc.getStructuringElement(Imgproc.MORPH_RECT, new Size(24, 24));
82          Mat erodeElement = Imgproc.getStructuringElement(Imgproc.MORPH_RECT, new Size(12, 12));
83  
84          Imgproc.erode(mask, morphOutput, erodeElement);
85          Imgproc.erode(mask, morphOutput, erodeElement);
86  
87          Imgproc.dilate(mask, morphOutput, dilateElement);
88          Imgproc.dilate(mask, morphOutput, dilateElement);
89  
90          // show the partial output
91          collector.ifPresent(coll -> coll.morph(morphOutput));
92  
93          // find the tennis ball(s) contours and show them
94          return this.detect(morphOutput, frame);
95      }

97      /**
98       * Given a binary image containing one or more closed surfaces, use it as a
99       * mask to find and highlight the objects contours
100      *
101      * @param maskedImage the binary image to be used as a mask
102      * @param frame       the original {@link Mat} image to be used for drawing the
103      *                    objects contours
104      * @return the {@link Mat} image with the objects contours framed
105      */
106     private StartLight detect(Mat maskedImage, Mat frame) {
107         // init
108         List<MatOfPoint> contours = new ArrayList<>();
109         Mat hierarchy = new Mat();
110 
111         // find contours
112         Imgproc.findContours(maskedImage, contours, hierarchy, Imgproc.RETR_CCOMP, Imgproc.CHAIN_APPROX_SIMPLE);
113 
114         StartLight startLight = this.startLight;
115         // if any contour exist...
116         if (hierarchy.size().height > 0 && hierarchy.size().width > 0) {
117             // for each contour, display it in blue
118             for (int idx = 0; idx >= 0; idx = (int) hierarchy.get(0, idx)[0]) {
119 
120 //                Imgproc.drawMarker(frame,   ,  new Scalar(250, 0, 0));
121                 // Imgproc.drawContours(frame, contours, idx, new Scalar(0, 0, 255));
122                 Imgproc.drawContours(frame, contours, idx, new Scalar(10, 250, 20), 10);
123             }
124 
125             startLight = startLight.on();
126         } else {
127             startLight = startLight.off();
128         }
129 
130         collector.ifPresent(coll -> coll.startLight(frame));
131         return startLight;
132     }
