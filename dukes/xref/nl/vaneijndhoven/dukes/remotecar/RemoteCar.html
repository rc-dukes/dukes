26    private static final Logger LOG = LoggerFactory.getLogger(RemoteCar.class);
27  
28    /**
29     * start me with given command line parameters
30     * 
31     * @param args
32     *          - the command line parameters to use
33     * @throws Exception
34     */
35    public static void main(String... args) throws Exception {
36      ClusterStarter starter = new ClusterStarter();
37      starter.prepare();
38      Car car = new Car(new Engine(new EngineMap()),
39          new Steering(new SteeringMap()));
40  
41      configureShutdownHook(car);
42  
43      // Set wheel and speed to neutral.
44      car.stop();
45      // Command.stop();
46  
47      LOG.info(
48          "Firing up General Lee vert.x and core controller, tutu tu tu tu tutututu tututu...");
49  
50      VertxOptions options = starter.getOptions();
51  
52      if (Environment.getInstance().runningOnRaspberryPi()) {
53        LOG.info(
54            "Running on the Raspberry Pi, activating Vert.x clustering over the network.");
55        // activate clustering over the network on the right interface.
56        options.setClusterHost(Environment.getInstance().getPiAddress());
57      } else {
58        LOG.info(
59            "Not running on the Raspberry Pi, not activating Vert.x clustering over the network.");
60      }
61  
62      Vertx.clusteredVertx(options, resultHandler -> {
63        Vertx vertx = resultHandler.result();
64        DeploymentOptions deploymentOptions = new DeploymentOptions();
65        deploymentOptions.setWorker(true);
66        vertx.deployVerticle(new WatchDog(), deploymentOptions);
67        vertx.deployVerticle(new CarVerticle(), deploymentOptions);
68        /*
69         * vertx.deployVerticle(new Daisy(), deploymentOptions);
70         * vertx.deployVerticle(new Daisy(), deploymentOptions, async -> {
71         * 
72         * if (async.failed()) { return; }
73         * 
74         * // vertx.eventBus().send(Events.STREAMADDED.name(), new
75         * JsonObject().put("source",
76         * "file://Users/jpoint/Repositories/dukes/daisy/src/main/resources/videos/full_run.mp4"
77         * )); }); vertx.deployVerticle(new Luke(), async -> { if (async.failed())
78         * { return; }
79         * 
80         * // vertx.eventBus().send(Characters.LUKE.getCallsign() + ":" +
81         * Luke.START_DRAG_NAVIGATION,null); });
82         */
83        // vertx.deployVerticle(new UncleJesse());
84      });
85    }
86  
87    private static void configureShutdownHook(Car car) {
88      Runtime.getRuntime().addShutdownHook(new Thread() {
89        public void run() {
90          LOG.info("Activating shutdown hook.");
91          car.stop();
92          Command.statusLedOff();
93        }
94      });
95    }
