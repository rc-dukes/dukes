28      /**
29       * start me with given command line parameters
30       * @param args - the command line parameters to use
31       * @throws Exception
32       */
33      public static void main(String... args) throws Exception {
34          Car car = new Car(new Engine(new EngineMap()), new Steering(new SteeringMap()));
35  
36          Config.configureLogging();
37          configureShutdownHook(car);
38  
39          // Set wheel and speed to neutral.
40          car.stop();
41  //        Command.stop();
42  
43          LOG.info("Firing up General Lee vert.x and core controller, tutu tu tu tu tutututu tututu...");
44  
45          VertxOptions options = new VertxOptions()
46                  .setClustered(true)
47                  .setClusterManager(Config.createHazelcastConfig());
48  
49          if (Environment.getInstance().runningOnRaspberryPi()) {
50              LOG.info("Running on the Raspberry Pi, activating Vert.x clustering over the network.");
51              // activate clustering over the network on the right interface.
52              options.setClusterHost(Environment.getInstance().getPiAddress());
53          } else {
54              LOG.info("Not running on the Raspberry Pi, not activating Vert.x clustering over the network.");
55          }
56  
57          Vertx.clusteredVertx(options, resultHandler -> {
58              Vertx vertx = resultHandler.result();
59              DeploymentOptions deploymentOptions = new DeploymentOptions();
60              deploymentOptions.setWorker(true);
61              vertx.deployVerticle(new Flash(), deploymentOptions);
62              vertx.deployVerticle(new Bo(), deploymentOptions);

84      private static void configureShutdownHook(Car car) {
85          Runtime.getRuntime().addShutdownHook(new Thread(){
86              public void run() {
87                  LOG.info("Activating shutdown hook.");
88                  car.stop();
89                  Command.statusLedOff();
90              }
91          });
92      }
