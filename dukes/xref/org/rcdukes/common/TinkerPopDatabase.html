8   import org.apache.tinkerpop.gremlin.process.traversal.IO;
9   import org.apache.tinkerpop.gremlin.process.traversal.dsl.graph.GraphTraversalSource;
10  import org.apache.tinkerpop.gremlin.structure.T;

13  
14  import io.vertx.core.json.JsonObject;
15  import io.vertx.core.logging.Logger;
16  import io.vertx.core.logging.LoggerFactory;
17  
18  /**
19   * a Gremlin/TinkerPop based in memory database
20   * 
21   * @author wf
22   *
23   */
24  public class TinkerPopDatabase implements GraphDatabase {
25    protected static final Logger LOG = LoggerFactory
26        .getLogger(TinkerPopDatabase.class);
27    protected TinkerGraph graph;
28    protected boolean debug=false;
29  
30    /**
31     * construct me
32     */
33    public TinkerPopDatabase() {
34      graph = TinkerGraph.open();
35    }
36  
37    /**
38     * access to graph
39     * 
40     * @return the GraphTraversalSource
41     */
42    public GraphTraversalSource g() {
43      return this.graph.traversal();
44    }
45  
46    /**
47     * set a property of the given vertex
48     * 
49     * @param v
50     * @param name
51     *          - name of the property
52     * @param value
53     *          - value of the property
54     */
55    public void setProp(Vertex v, String name, Object value) {
56      if (value != null) {
57        v.property(name, value);
58      }
59    }
60  
61    /**
62     * write me to the given path
63     * 
64     * @param path
65     */
66    public void writeGraph(String path) {
67      this.g().io(path).with(IO.writer, IO.graphson).write().iterate();
68    }
69  
70    /**
71     * load the given graphFile
72     * 
73     * @param graphFile
74     */
75    public void loadGraph(File graphFile) {
76      this.g().io(graphFile.getPath()).with(IO.reader, IO.graphson).read()
77          .iterate();
78      long nodeCount = this.g().V().count().next().longValue();
79      if (debug) {
80        String msg = String.format("loaded graph %s with %d nodes",
81            graphFile.getPath(), nodeCount);
82        LOG.info(msg);
83      }
84    }
85  
86    /**
87     * add a vertex for the given object
88     * 
89     * @param o
90     */
91    public void addVertex(Object o) {
92      Vertex v = this.graph.addVertex(T.label, o.getClass().getSimpleName());
93      JsonObject jo = JsonObject.mapFrom(o);
94      for (Entry<String, Object> entry : jo.getMap().entrySet()) {
95        Object value=entry.getValue();
96        if (value!=null)
97          v.property(entry.getKey(), value);
98      }
99    }
100 
101   @Override
102   public <TYPE> TYPE fromVertex(Vertex v, Class<TYPE> clazz) {
103     Map<String, Object> map = new HashMap<String, Object>();
104     JsonObject jo = new JsonObject(map);
105     for (String key : v.keys()) {
106       jo.put(key, v.property(key).value());
107     }
108     return jo.mapTo(clazz);
109   }
110 }
