87    /**
88     * wait for me to be started
89     * 
90     * @param timeOut
91     *          - in msecs
92     * @param pollTime
93     *          - in msec
94     * @throws Exception
95     * @return the number of milliseconds the wait took
96     */
97    public int waitStatus(Status status, int timeOut, int pollTime)
98        throws Exception {
99      int leftTime = timeOut;
100     while (!this.status.equals(status)) {
101       Thread.sleep(pollTime);
102       leftTime -= pollTime;
103       if (leftTime < 0) {
104         String msg = String.format("wait %s timed out after %d msecs",
105             status.name(), timeOut);
106         throw new Exception(msg);
107       }
108     }
109     if (debug) {
110       String msg = String.format("wait %s success after %d msecs",
111           status.name(), timeOut - leftTime);
112       LOG.info(msg);
113     }
114     return timeOut - leftTime;
115   }

117   /**
118    * convert the given list of name Value to a JsonObject
119    * 
120    * @param nameValues
121    *          - needs to be an even number of arguments otherwise an
122    *          IllegalArgumentException runtime exception is thrown
123    * @return - the JsonObject create from the nameValues by putting name,value
124    *         into the object in a pairwise fassion
125    */
126   public JsonObject toJsonObject(String... nameValues) {
127     JsonObject jo = new JsonObject();
128     if (nameValues.length % 2 != 0) {
129       String msg = String.format("got %d namevalue but expected even number",
130           nameValues.length);
131       throw new IllegalArgumentException(msg);
132     }
133     for (int i = 0; i < nameValues.length; i += 2) {
134       String name = nameValues[i];
135       String value = nameValues[i + 1];
136       jo.put(name, value);
137     }
138     return jo;
139   }

237   /**
238    * share the data for the given jsonobject under the given key - if the json
239    * 
240    * @param key
241    *          - the key for the JsonObject
242    * @param jo
243    *          the json object
244    */
245   public void shareData(String key, JsonObject jo) {
246     if (jo == null)
247       return;
248     String msg = String.format("sharing %s: %s", key, jo.encodePrettily());
249     LOG.info(msg);
250     vertx.sharedData().getLocalMap(character.getCallsign()).put(key, jo);
251   }

278   /**
279    * construct a Pojo from the given JsonObject
280    * 
281    * @param jo
282    *          - the JsonObject
283    * @param clazz
284    *          - the type to use
285    * @return - the mapped instance or a default instance from a no-args
286    *         constructor if jo is null
287    */
288   public <T> T fromJsonObject(JsonObject jo, Class<T> clazz) {
289     T result;
290     if (jo == null) {
291       try {
292         result = clazz.newInstance();
293       } catch (InstantiationException | IllegalAccessException e) {
294         LOG.trace(e.getMessage());
295         result = null;
296       }
297     } else {
298       result = jo.mapTo(clazz);
299     }
300     return result;
301   }
