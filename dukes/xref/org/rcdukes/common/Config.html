68    /**
69     * configure the logging
70     */
71    public static void configureLogging() {
72      System.setProperty(
73          io.vertx.core.logging.LoggerFactory.LOGGER_DELEGATE_FACTORY_CLASS_NAME,
74          SLF4JLogDelegateFactory.class.getName());
75  
76      LoggerContext logConfig = (LoggerContext) LoggerFactory.getILoggerFactory();
77      logConfig.getLogger("ROOT").setLevel(Level.TRACE);
78      logConfig.getLogger("nl.vaneijndhoven.dukes.imageview").setLevel(Level.INFO);
79      logConfig.getLogger("io.vertx").setLevel(Level.INFO);
80      logConfig.getLogger("com.hazelcast").setLevel(Level.INFO);
81      logConfig.getLogger("com.hazelcast.nio.tcp").setLevel(Level.ERROR);
82      logConfig.getLogger("io.netty").setLevel(Level.INFO);
83    }

85    /**
86     * create a ClusterManager with a default hazelcast configuration using slf4j
87     * 
88     * @return - the ClusterManager
89     */
90    public static ClusterManager createHazelcastConfig() {
91      com.hazelcast.config.Config hazelcastConfig = new com.hazelcast.config.Config();
92      hazelcastConfig.setProperty("hazelcast.logging.type", "slf4j");
93      ClusterManager mgr = new HazelcastClusterManager(hazelcastConfig);
94      return mgr;
95    }

97    /**
98     * get my ip addresses
99     * 
100    * @return - my ip addresses
101    */
102   public static List<String> getMyIpAddresses() {
103     try {
104       return Collections.list(NetworkInterface.getNetworkInterfaces()).stream()
105           .map(iface -> Collections.list(iface.getInetAddresses()))
106           .flatMap(Collection::stream).map(InetAddress::getHostAddress)
107           .collect(Collectors.toList());
108     } catch (SocketException e) {
109       LOG.error("Error while determining IP addresses: ", e);
110       return null;
111     }
112   }
