1   package org.rcdukes.car;

3   import org.rcdukes.common.Characters;
4   import org.rcdukes.common.DukesVerticle;
5   import org.rcdukes.common.Events;
6   import org.rcdukes.common.ServoPosition;
7   import org.rcdukes.drivecontrol.Car;
8   
9   import io.vertx.core.json.JsonObject;
10  import io.vertx.rxjava.core.eventbus.Message;
11  import rx.Subscription;
12  
13  /**
14   * Motor and steering output aka Bo
15   */
16  public class CarVerticle extends DukesVerticle {
17  
18    /**
19     * start the car verticle
20     */
21    public CarVerticle() {
22      super(Characters.BO);
23    }
24  
25    private Car car = Car.getInstance(); // there is only one car for the time
26                                         // being
27    private SpeedHandler speedHandler = new SpeedHandler(car);
28    private SteeringHandler steeringHandler = new SteeringHandler(car);
29  
30    @Override
31    public void start() {
32      super.preStart();
33      Subscription subscription = vertx.eventBus()
34          .consumer(Characters.BO.getCallsign()).toObservable()
35          .doOnNext(x -> LOG.trace("Received instruction")).map(Message::body)
36          .cast(JsonObject.class).subscribe(message -> handleCarMessage(message));
37      super.postStart();
38    }
39  
40    private void handleCarMessage(JsonObject message) {
41      LOG.trace("Instruction: {}", message);
42      String type = message.getString("type");
43      ServoPosition newPosition = null;
44      switch (type) {
45      case "motor":
46        newPosition = speedHandler.handleMotor(message);
47        newPosition.kind="motor";
48        break;
49      case "servo":
50        newPosition = steeringHandler.handleServo(message);
51        newPosition.kind="steering";
52        break;
53      case "servoAngle":
54        newPosition=steeringHandler.handleServoAngle(message);
55        newPosition.kind="steering";
56        break;
57      case "servoDirect":
58        newPosition = steeringHandler.handleServoDirect(message);
59        newPosition.kind="steering";
60        break;
61      case "speedDirect":
62        newPosition = speedHandler.handleSpeedDirect(message);
63        newPosition.kind="motor";
64        break;
65      case "log":
66        String logMessage = message.getString("message");
67        LOG.debug("Received log message: " + logMessage);
68        break;
69      default:
70        LOG.error("Unknown message type {}", type);
71      }
72      if (newPosition!=null) {
73        JsonObject posReply=JsonObject.mapFrom(newPosition);
74        this.sendEvent(Characters.BOSS_HOGG, Events.CAR_POSITION, posReply);
75      }
76  
77    }
78  }
