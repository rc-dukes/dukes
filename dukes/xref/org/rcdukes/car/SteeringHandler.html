1   package org.rcdukes.car;
2   
3   import org.rcdukes.common.Config;
4   import org.rcdukes.common.ServoPosition;
5   import org.rcdukes.drivecontrol.Car;
6   
7   import io.vertx.core.json.JsonObject;
8   import io.vertx.core.logging.Logger;
9   import io.vertx.core.logging.LoggerFactory;
10  
11  /**
12   * handle the steering of the car from given command messages
13   *
14   */
15  class SteeringHandler {
16  
17    private static final Logger LOG = LoggerFactory
18        .getLogger(SteeringHandler.class);
19  
20    private Car car;
21    private ServoRangeMap steeringMap;
22  
23    private int steerFactor;
24    
25    /**
26     * create a steering handler for the given car
27     * 
28     * @param car
29     */
30    SteeringHandler(Car car) {
31      this.car = car;
32      this.steeringMap=car.getSteering().getSteeringMap(); 
33      steerFactor=steeringMap.turnedOrientation()?-1:1;
34    }
35  
36    /**
37     * handle direct servo Messages - the position is to be absolutely set
38     * 
39     * @param messageBody
40     *          containing a double value "position" element
41     * @return the new servo position
42     */
43    ServoPosition handleServoDirect(JsonObject messageBody) {
44      LOG.debug("Received direct message for servo: " + messageBody);
45      String position = messageBody.getString("position");
46      int percent = Double.valueOf(position).intValue()*steerFactor;
47      ServoPosition newPosition=this.steeringMap.atPercent(percent);
48      car.turn(newPosition);
49      return newPosition;
50    }
51    
52    /**
53     * handle the given servo Angle
54     * @param angleJo
55     * @return
56     */
57    public ServoPosition handleServoAngle(JsonObject angleJo) {
58      Double angle = angleJo.getDouble("angle");
59      ServoPosition newPosition=this.steeringMap.atValue(angle*steerFactor);
60      String msg=String.format("steering %5.1f° for wanted angle %5.1f°", newPosition.getValue(),angle);
61      LOG.debug(msg);
62      car.turn(newPosition);
63      return newPosition;
64    }
65  
66    /**
67     * handle a message for the steering servo
68     * 
69     * @param messageBody
70     *          - containing a position element with "left"/"right" commands
71     * @return 
72     */
73    ServoPosition handleServo(JsonObject messageBody) {
74      LOG.debug("Received message for servo: " + messageBody);
75      String position = messageBody.getString("position");
76      if (Config.POSITION_LEFT.equals(position)) {
77        steeringMap.step(-1*steerFactor);
78      } else if (Config.POSITION_RIGHT.equals(position)) {
79        steeringMap.step(1*steerFactor);
80      } else if (Config.POSITION_CENTER.equals(position)) {
81        steeringMap.setZero();
82      }
83      ServoPosition newPosition=steeringMap.getCurrentPosition();
84      car.turn(newPosition);
85      return newPosition;
86    }
87  
88  }
