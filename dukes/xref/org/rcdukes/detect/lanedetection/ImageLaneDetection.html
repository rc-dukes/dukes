32  /**
33   * image lane detection
34   *
35   */
36  public class ImageLaneDetection {
37    protected static final Logger LOG = LoggerFactory
38        .getLogger(ImageLaneDetection.class);
39    private LaneDetector ld;
40  
41    /**
42     * create me for the given LaneDetector
43     * 
44     * @param laneDetector
45     */
46    public ImageLaneDetection(LaneDetector laneDetector) {
47      this.ld = laneDetector;
48    }
49  
50    /**
51     * detect the lane
52     * 
53     * @param image
54     * @param imageCollector
55     * @return a map with information
56     */
57    public LaneDetectionResult detectLane(Image image,
58        ImageCollector imageCollector) {
59      LaneDetectionResult ldr = new LaneDetectionResult();
60      if (image == null) {
61        LOG.error("detectLane: original is null");
62        return ldr;
63      }
64      CameraConfig cameraConfig = ld.getCameraConfig();
65      imageCollector.addImage(image, ImageType.camera);
66      ldr.frameIndex = image.getFrameIndex();
67      ldr.milliTimeStamp = image.getMilliTimeStamp();
68      // ! important - create a copy of the image because debug info might be
69      // written on it
70      Mat imageCopy = image.getFrame().clone();
71      Mat undistorted = ld.getMatrix().apply(imageCopy);
72      // get the configured Region of interest
73      double ry = cameraConfig.getRoiy() / 100.0;
74      double rh = (1 - ry) * cameraConfig.getRoih() / 100.0;
75      Mat frame = new ROI("camera", 0, ry, 1, rh).region(undistorted);
76      Size imageSize = frame.size();
77      ViewPort viewPort = new ViewPort(new Point(0.0, 0.0), imageSize.width,
78          imageSize.height);
79      Polygon imagePolygon = new ImagePolygon(frame.size(), 0, 0, 1, 0, 1, 1, 0,
80          1);
81      Polygon worldPolygon = new ImagePolygon(frame.size(), 0, 0, 1, 0, 1, 1, 0,
82          1);
83  
84      PerspectiveShift perspectiveShift = new PerspectiveShift(imagePolygon,
85          worldPolygon);
86      Mat birdseye = perspectiveShift.apply(frame);
87      imageCollector.addImage(birdseye, ImageType.birdseye);
88  
89      // step1 edge detection
90      Mat imgEdges = ld.getEdgeDetector().detect(frame);
91      imageCollector.addImage(imgEdges, ImageType.edges);
92  
93      // step 2 line detection
94      Collection<Line> lines = ld.getLineDetector().detect(imgEdges);
95      Lane lane = new DefaultLaneDetector().detect(lines, viewPort);
96      StoppingZone stoppingZone = new DefaultStoppingZoneDetector().detect(lines);
97  
98      LaneOrientation laneOrientation = new LaneOrientation(lane, viewPort);
99      laneOrientation.determineLines();
100     StoppingZoneOrientation stoppingZoneOrientation = new StoppingZoneOrientation(
101         stoppingZone, lane, viewPort);
102 
103     Optional<Line> middle = Optional.ofNullable(laneOrientation.getMiddle());
104 
105     ImageUtils iu = new ImageUtils();
106     lane.getLeftBoundary().ifPresent(boundary -> iu.drawLinesToImage(frame,
107         asList(boundary), CVColor.green));
108     lane.getRightBoundary().ifPresent(boundary -> iu.drawLinesToImage(frame,
109         asList(boundary), CVColor.dodgerblue));
110     middle.ifPresent(
111         line -> iu.drawLinesToImage(frame, asList(line), CVColor.red));
112 
113     ldr.distanceToStoppingZone = -1.0;
114     ldr.distanceToStoppingZoneEnd = -1.0;
115     if (stoppingZone.getEntrance() != null) {
116       if (cameraConfig.isShowStoppingZone())
117         stoppingZone.getEntrance().ifPresent(entrance -> iu
118             .drawLinesToImage(frame, asList(entrance), CVColor.cyan));
119       ldr.distanceToStoppingZone = stoppingZoneOrientation
120           .determineDistanceToStoppingZone();
121     }
122 
123     if (stoppingZone.getExit() != null) {
124       if (cameraConfig.isShowStoppingZone())
125         stoppingZone.getExit().ifPresent(
126             exit -> iu.drawLinesToImage(frame, asList(exit), CVColor.yellow));
127       ldr.distanceToStoppingZoneEnd = stoppingZoneOrientation
128           .determineDistanceToStoppingZoneEnd();
129     }
130     imageCollector.addImage(frame, ImageType.lines);
131 
132     Double angleOffset = cameraConfig.getAngleOffset();
133     if (laneOrientation.getLeft() != null)
134       ldr.left = laneOrientation.getLeft().angleDeg90() + angleOffset;
135     if (laneOrientation.getMiddle() != null)
136       ldr.middle = laneOrientation.getMiddle().angleDeg90() + angleOffset;
137     if (laneOrientation.getRight() != null)
138       ldr.right = laneOrientation.getRight().angleDeg90() + angleOffset;
139     ldr.distanceMiddle = laneOrientation.determineDistanceToMiddle();
140     ldr.distanceLeft = laneOrientation.distanceFromLeftBoundary();
141     ldr.distanceRight = laneOrientation.distanceFromRightBoundary();
142     ldr.courseRelativeToHorizon = laneOrientation
143         .determineCourseRelativeToHorizon();
144     return ldr;
145   }
