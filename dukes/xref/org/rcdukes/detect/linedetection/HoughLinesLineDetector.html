12  /**
13   * Hough Lines Line Detector
14   *
15   */
16  public class HoughLinesLineDetector implements LineDetector {
17  
18    double rho = 1;
19    double theta = Math.PI / 180; // By choosing this value lines sloping
20                                  // left to right will be < 0 radian,
21                                  // while lines sloping right to left
22                                  // will be > 0 radian.
23    int threshold = 70;
24    double minLineLength = 20;
25    double maxLineGap = 10;
26    boolean probabilistic = true;
27   
28    /**
29     * default constructor
30     */
31    public HoughLinesLineDetector() {
32      
33    }
34    
35    /**
36     * construct me
37     * 
38     * @param rho
39     * @param theta
40     * @param threshold
41     * @param minLineLength
42     * @param maxLineGap
43     */
44    public HoughLinesLineDetector(boolean probabilistic, double rho, double theta,
45        int threshold, double minLineLength, double maxLineGap) {
46      this.probabilistic = probabilistic;
47      this.rho = rho;
48      this.theta = theta;
49      this.threshold = threshold;
50      this.minLineLength = minLineLength;
51      this.maxLineGap = maxLineGap;
52    }
53  
54    @Override
55    public Collection<Line> detect(Mat image) {
56      Mat houghLinesMat=detectMat(image);
57      
58      
59      Set<Line> lines = new HashSet<Line>();
60      for (int x = 0; x < houghLinesMat.rows(); x++) {
61        Line line = Line.fromHough(probabilistic,image.width(),image.height(),houghLinesMat.get(x, 0));
62        lines.add(line);
63      }
64      houghLinesMat.release();
65      return lines;
66    }
67  
68    /**
69     * detect lines
70     * 
71     * @param image
72     * @return a Mat with encoded lines see 
73     * <a href="https://docs.opencv.org/3.4/d9/db0/tutorial_hough_lines.html">OpenCV tutorial hough lines</a>
74     */
75    private Mat detectMat(Mat image) {
76      Mat lineMat = new Mat();
77      // see 
78      if (probabilistic) {
79        Imgproc.HoughLinesP(image, lineMat, rho, theta, threshold, minLineLength,
80            maxLineGap);
81      } else {
82        Imgproc.HoughLines(image, lineMat, rho, theta, threshold);
83      }
84      return lineMat;
85    }
