44    // @TODO Make configurable
45    // the format to be used for image encoding

48      png, jpg

51    public static ImageFormat imageFormat = ImageFormat.jpg;
52    public static String exts[] = { ".png", ".jpg" };
53    public static String contentTypes[] = { "image/png", "image/jpeg" };
54    // @TODO Make configurable and adapt
55    public static double fps = 10;
56  
57    @Override
58    public void start(Future<Void> startFuture) throws Exception {
59      super.preStart();
60      // @TODO - get config information from config Verticle (or shared data ...)
61      File mediaPath=new File(Environment.dukesHome+"media");
62      mediaPath.mkdirs();
63      ImageUtils.MEDIA_PATH=mediaPath.getPath();
64      int port = Config.getEnvironment().getInteger(Config.IMAGEVIEW_PORT);
65      this.videoRecorders=new VideoRecorders(fps);
66      server = vertx.createHttpServer().requestHandler(this::sendImage);
67      // Now bind the server:
68      server.listen(port, res -> {
69        if (res.succeeded()) {
70          startFuture.complete();
71          consumer(Events.START_RECORDING, x -> videoRecorders.start());
72          consumer(Events.STOP_RECORDING, x -> videoRecorders.stop());
73          consumer(Events.SIMULATOR_IMAGE, this::receiveSimulatorImage);
74          consumer(Events.PHOTO_SHOOT,x->shootPhoto());
75          super.postStart();
76          String msg = String.format(
77              "web server on port %d serving debug images in %s format", port,
78              exts[imageFormat.ordinal()]);
79          LOG.info(msg);
80        } else {
81          startFuture.fail(res.cause());
82        }
83      });
84    }
85    
86    /**
87     * receive an image from the simulator
88     */
89    protected void receiveSimulatorImage(Message<String> message) {
90      String imgData = message.body(); // in DataURL format ...
91      try {
92        Mat imageMat=ImageUtils.matFromDataUrl(imgData, "jpg");
93        ImageCollector imageCollector = Detector.getImageCollector();
94        imageCollector.addImage(imageMat, ImageType.simulator);
95      } catch (IOException e) {
96        ErrorHandler.getInstance().handle(e);
97      }
98    }

119   /**
120    * send an image for the given request
121    * 
122    * @param request
123    */
124   public void sendImage(HttpServerRequest request) {
125     String type = request.getParam("type");
126     String mode = request.getParam("mode");
127     if (type == null)
128       type = "camera";
129     ImageType imageType = ImageType.valueOf(type);
130     if (mode != null && mode.equals("stream")) {
131       sendStream(request, imageType);
132     } else {
133       sendSingleImage(request, imageType);
134     }
135   }
136 
137   /**
138    * send a single Image
139    * 
140    * @param request
141    * @param imageType
142    */
143   public void sendSingleImage(HttpServerRequest request, ImageType imageType) {
144     Mat mat = null;
145     Image image=this.getNextImage(imageType);
146     if (image != null) {
147       mat = image.getFrame();
148       this.sendImageBytes(request, image.getImageBytes());
149     }
150     // optionally record
151     videoRecorders.recordFrame(mat,imageType);
152   }
153   
154   /**
155    * send the given bytes as an image
156    * 
157    * @param bytes
158    */
159   public void sendImageBytes(HttpServerRequest request, byte[] bytes) {
160     HttpServerResponse response = request.response();
161     String contentType = contentTypes[imageFormat.ordinal()];
162     response.putHeader("content-type", contentType);
163     response.putHeader("content-length", "" + bytes.length);
164     Buffer data = Buffer.buffer().appendBytes(bytes);
165     response.write(data);
166     response.close();
167   }
168 
169   /**
170    * stream images
171    * 
172    * @param request
173    * @param imageType
174    */
175   private void sendStream(HttpServerRequest request, ImageType imageType) {
176     HttpServerResponse response = request.response();
177     Runnable multipartStreamer = new MultipartStreamer(this, response, imageType);
178     Thread streamThread = new Thread(multipartStreamer);
179     streamThread.start();
180   }

183    * delegator
184    * @param frame
185    * @param imageType

187   public void recordFrame(Mat frame, ImageType imageType) {
188     this.videoRecorders.recordFrame(frame,imageType);
189   }
