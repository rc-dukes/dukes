17  /**
18   * reactive MJPegDecoder
19   * 
20   * @author wf
21   *
22   */
23  public class MJpegDecoder extends Subscriber<byte[]> {
24    protected static final Logger LOG = LoggerFactory
25        .getLogger(MJpegDecoder.class);
26  
27    int prev = 0;
28    int cur = 0;
29    private ByteArrayOutputStream jpgOut;
30    private byte[] curFrame;
31    public static boolean debug = false;
32    private int bufferIndex = 0;
33    private int frameIndex = 0;
34    FileOutputStream fos;
35  
36    private int bufferSize;
37  
38    private Observable<byte[]> mjpegSubscription;
39  
40    private MJpegHandler mjpegHandler;
41  
42    /**
43     * open the decoder for the given stream
44     * 
45     * @param mJpegHandler
46     * @param bufferSize
47     */
48    public MJpegDecoder(MJpegHandler mJpegHandler) {
49      this.mjpegHandler = mJpegHandler;
50      this.curFrame = new byte[0];
51      if (debug) {
52        try {
53          fos = new FileOutputStream("/tmp/decoder.mjpg");
54        } catch (FileNotFoundException e) {
55          handle(e);
56        }
57      }
58    }
59  
60    @Override
61    public void onCompleted() {
62      try {
63        this.mjpegHandler.close();
64        if (fos != null) {
65          fos.close();
66        }
67        fos = null;
68      } catch (IOException e) {
69        onError(e);
70      }
71    }
72  
73    @Override
74    public void onError(Throwable e) {
75      handle(e);
76    }
77  
78    @Override
79    public void onNext(byte[] buffer) {
80      if (debug) {
81        String msg = String.format("buffer %6d available %9d kB read",
82            ++bufferIndex, bufferIndex * bufferSize / 1024);
83        LOG.info(msg);
84        try {
85          fos.write(curFrame);
86          fos.flush();
87        } catch (IOException e) {
88          handle(e);
89        }
90  
91      }
92      // loop over all bytes in the buffer
93      for (int cur : buffer) {
94        // Content-Type: multipart/x-mixed-replace; boundary=
95        // will have -- we could detect it here
96        if (debug) {
97          if (prev == 0x2D && cur == 0x2D) {
98            LOG.info("boundary detected");
99          }
100       }
101       // check for JPEG start bytes
102       if (prev == 0xFFFFFFFF && cur == 0xFFFFFFD8) {
103         if (debug) {
104           String msg = String.format("frame %6d started", frameIndex + 1);
105           LOG.info(msg);
106         }
107         jpgOut = new ByteArrayOutputStream(bufferSize);
108         // first byte needs to be written to output
109         jpgOut.write((byte) prev);
110       }
111       // if within the frame write all bytes
112       if (jpgOut != null) {
113         jpgOut.write((byte) cur);
114         // check for JPEG end bytes
115         // if found the frame is finished
116         if (prev == 0xFFFFFFFF && cur == 0xFFFFFFD9) {
117           // create the byte array of the current jpeg frame
118           curFrame = jpgOut.toByteArray();
119           try {
120             jpgOut.close();
121             jpgOut = null;
122           } catch (IOException e) {
123             onError(e);
124           }
125 
126           if (debug) {
127             String msg = String.format("frame %6d available", ++frameIndex);
128             LOG.info(msg);
129           }
130 
131           // emit the current frame
132         }
133       }
134       prev = cur;
135     }
136   }
137 
138   private void handle(Throwable th) {
139     LOG.error(th.getMessage());
140     if (debug)
141       th.printStackTrace();
142   }
143 
144   /**
145    * open me with the given bufferSize
146    * 
147    * @param bufferSize
148    *          e.g. 64 KByte Buffer - 10.5 msecs/100 FPS at 1920x1080
149    *          1000/(1920*1080*3/1024/64)
150    */
151   public void open(int bufferSize) {
152     this.bufferSize = bufferSize;
153     BodyDeferringInputStream inputStream = this.mjpegHandler.getInputStream();
154     if (inputStream != null) {
155       mjpegSubscription = StringObservable.from(inputStream, bufferSize)
156           .subscribeOn(Schedulers.io()).observeOn(Schedulers.newThread());
157       mjpegSubscription.subscribe(this);
158     }
159   }
160 
161   /**
162    * close me
163    */
164   public void close() {
165     this.unsubscribe();
166     this.onCompleted();
167   }
