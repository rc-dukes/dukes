82    /**
83     * convert the given lane Detection Result to a vertex and add it to the graph
84     * 
85     * @param ldr
86     * @return the new Vertex
87     */
88    public Vertex addToGraph(LaneDetectionResult ldr) {
89      Vertex v = graph.addVertex(T.label, "pos", "frameIndex", ldr.frameIndex,
90          "milliTimeStamp", ldr.milliTimeStamp);
91      setProp(v, "left", ldr.left);
92      setProp(v, "middle", ldr.middle);
93      setProp(v, "right", ldr.right);
94      setProp(v, "course", ldr.courseRelativeToHorizon);
95      return v;
96    }

111   /**
112    * is it o.k to send a command with the given AngleRange?
113    * 
114    * @param angleRange
115    *          - the angle Range to check
116    * @param minFound
117    * @return true if the angleRange is not null has enough found entries and
118    *         enough time has passed since the latest command
119    */
120   public boolean cmdOk(AngleRange angleRange, int minFound) {
121     boolean ok = angleRange != null && angleRange.count > minFound
122         && currentTime - tsLatestCommand > COMMAND_LOOP_INTERVAL;
123     return ok;
124   }

126   /**
127    * staticical analysis of angles in the given past timeWindow
128    * 
129    * @author wf
130    *
131    */
132   class AngleRange {
133     String name;
134     int timeWindow;
135     
136     int count;
137     double sum = 0;
138     
139     Double min = Double.MAX_VALUE;
140     Double max = -Double.MAX_VALUE;
141     Double avg = null;
142     Double stdDev=null;
143 
144     /**
145      * create an angle Range for the given vertices
146      * 
147      * @param name
148      * @param currentTime
149      * @param timeWindow
150      */
151     public AngleRange(String name, long currentTime, int timeWindow) {
152       this.name = name;
153       List<Vertex> angleNodes = g().V().has(name)
154           .has("milliTimeStamp", P.gt(currentTime - timeWindow)).toList();
155       count = angleNodes.size();
156       this.timeWindow = timeWindow;
157       if (count == 0) {
158         min = null;
159         max = null;
160       }
161 
162       for (Vertex angleNode : angleNodes) {
163         double angle = (Double) angleNode.property(name).value();
164         sum += angle;
165         max = Math.max(max, angle);
166         min = Math.min(min, angle);
167       }
168       avg = sum / count;
169       double sum2 = 0;
170       for (Vertex angleNode : angleNodes) {
171         Double angle = (Double) angleNode.property(name).value();
172         Double avgDiff = angle - avg;
173         sum2 += avgDiff * avgDiff;
174       }
175       stdDev = Math.sqrt(sum2 / count);
176     }
177 
178     /**
179      * return the value to steer with for this angle range
180      * @return - the steering value
181      */
182     public double steer() {
183       double angle = avg;
184       // @TODO check polarity and factor
185       if (!"course".equals(name)) {
186         angle=-0.5*angle;
187       }
188       return angle;
189     }
190 
191     /**
192      * get the debug info
193      * 
194      * @return
195      */
196     public String debugInfo() {
197       String info = String.format("%6s: %s <- %s ± %s -> %s / %3d in %3d msecs",
198           name, Line.angleString(min), Line.angleString(avg),
199           Line.angleString(stdDev), Line.angleString(max), count, timeWindow);
200       return info;
201     }
202   }

217   /**
218    * analyze the angle ranges int the given timeWindow
219    * 
220    * @param showDebug
221    */
222   public void analyzeAngleRanges(int timeWindow, boolean showDebug) {
223     long aStart = System.nanoTime();
224     middleRange = new AngleRange("middle", currentTime, timeWindow);
225     stopRangeLeft = new AngleRange("left", currentTime,
226         MAX_DURATION_NO_LINES_DETECTED);
227     stopRangeRight = new AngleRange("right", currentTime,
228         MAX_DURATION_NO_LINES_DETECTED);
229     leftRange = new AngleRange("left", currentTime, timeWindow);
230     rightRange = new AngleRange("right", currentTime, timeWindow);
231     courseRange = new AngleRange("course", currentTime, timeWindow);
232     if (showDebug) {
233       long aTime = System.nanoTime()-aStart;
234       LOG.info(String.format("analysis took %5.3f msecs", aTime/1000000.0));
235       LOG.info(stopRangeLeft.debugInfo());
236       LOG.info(stopRangeRight.debugInfo());
237       LOG.info(middleRange.debugInfo());
238       LOG.info(leftRange.debugInfo());
239       LOG.info(rightRange.debugInfo());
240       LOG.info(courseRange.debugInfo());
241     }
242   }

244   /**
245    * process the laneDetectResult
246    * 
247    * @param ldr
248    *          - the lane detection result
249    * @return - a message to be sent to the vehicle or null on error
250    */
251   @Override
252   public JsonObject getNavigationInstruction(LaneDetectionResult ldr) {
253     // empty message signals no navigation
254     JsonObject message = null;
255 
256     // if in emergency stop mode do not continue
257     if (this.emergencyStopActivated)
258       return message;
259     // set the current time and start time
260     setTime(ldr);
261     // add the lane detection result to the tinker graph
262     Vertex ldrVertex=addToGraph(ldr);
263     // analyze the LaneDetectionResults of the relevant time Windows
264     boolean showDebug = true;
265     analyzeAngleRanges(timeWindow, showDebug);
266 
267     // do we need to stop since there have been no lines for too long?
268     if (currentTime - startTime > MAX_DURATION_NO_LINES_DETECTED
269         && stopRangeLeft.count + stopRangeRight.count == 0
270         || emergencyStopActivated) {
271       this.emergencyStopActivated = true;
272       return ActionVerticle.emergencyStopCommand();
273     }
274     // if we have enough detections for statistics we'll use this information
275     int MIN_FOUND_PER_TIMEWINDOW = 3;
276     // decide which angleRange to use
277     AngleRange navigateRange = null;
278     if (middleRange.count >= MIN_FOUND_PER_TIMEWINDOW) {
279       navigateRange = courseRange;
280     } else {
281       // are we left or right heavy?
282       navigateRange = leftRange.count > rightRange.count ? leftRange
283           : rightRange;
284     }
285     // check that we can send a command
286     if (this.cmdOk(navigateRange, MIN_FOUND_PER_TIMEWINDOW)) {
287       double angle = navigateRange.steer();
288       String msg = ldr.debugInfo() + String.format("\nsteer by %s: %s",
289           navigateRange.name, Line.angleString(angle));
290       LOG.debug(msg);
291       ldrVertex.property("steer",angle);
292       ldrVertex.property("steerBy",navigateRange.name);
293       ldrVertex.property("debugInfo",msg);
294       tsLatestCommand = currentTime;
295       message = steerCommand(angle);
296     }
297     return message;
298   }

300   /**
301    * get the command to steer the vehicle
302    * 
303    * @param angle
304    * @return - the command message
305    */
306   public JsonObject steerCommand(Double angle) {
307     String angleStr = String.format(Locale.ENGLISH, "%5.1f°", angle);
308     JsonObject message = new JsonObject().put("type", "servoAngle").put("angle",
309         angle);
310     String debugMsg = String.format("sending servoAngle %s", angleStr);
311     if (debug)
312       LOG.debug(debugMsg);
313     return message;
314   }
