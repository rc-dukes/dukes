29    /**
30     * get a LaneDetectionResult for the given Parameters
31     * 
32     * @param frameIndex
33     * @param timeStamp
34     * @param left
35     * @param middle
36     * @param right
37     * @return
38     */
39    public LaneDetectionResult getLdr(int frameIndex, int timeStamp, Double left,
40        Double middle, Double right) {
41      LaneDetectionResult ldr = new LaneDetectionResult();
42      ldr.frameIndex = frameIndex;
43      ldr.milliTimeStamp = timeStamp;
44      ldr.left = left;
45      ldr.middle = middle;
46      ldr.right = right;
47      return ldr;
48    }
49  
50    @Test
51    public void testStraightLaneNavigator() {
52      LaneDetectionResult ldrs[] = { getLdr(1, 0, -45., 0., 45.),
53          getLdr(2, 100, -46., 1., 46.), getLdr(3, 200, -47., 0., 47.) };
54      StraightLaneNavigator nav = new StraightLaneNavigator();
55      for (LaneDetectionResult ldr : ldrs)
56        nav.getNavigationInstruction(ldr);
57      long nodeCount = nav.g().V().count().next().longValue();
58      assertEquals(3, nodeCount);
59    }
60  
61    @Test
62    public void testFromJson() {

64      JsonObject ldrJo = new JsonObject();
65      LaneDetectionResult ldr = navigator.fromJsonObject(ldrJo);
66      JsonObject nav = navigator.getNavigationInstruction(ldr);
67      assertNull(nav);

70    @Test
71    public void testFromVideo() throws Exception {
72      NativeLibrary.load();
73      Navigator nav = new StraightLaneNavigator();
74      String testUrl = "http://wiki.bitplan.com/videos/full_run.mp4";
75      ImageFetcher imageFetcher = new ImageFetcher(testUrl);
76      imageFetcher.open();
77      int frameIndex = 0;
78      int maxFrameIndex = 300;
79      while (frameIndex < maxFrameIndex && imageFetcher.hasNext()
80          && !imageFetcher.isClosed()) {
81        Image image = imageFetcher.fetch();
82        frameIndex = image.getFrameIndex();
83        System.out.println(frameIndex);
84        if (imageFetcher.hasNext()) {
85          LaneDetector laneDetector = LaneDetector.getDefault();
86          LaneDetectionResult ldr = laneDetector.detect(image);
87          if (debug)
88          System.out.println(ldr.debugInfo());
89          JsonObject navJo = nav.getNavigationInstruction(ldr);
90          if (navJo != null)
91            System.out.println(navJo.encodePrettily());
92        }
93      }
94      imageFetcher.close();
95      long nodeCount = nav.g().V().count().next().longValue();
96      assertEquals(maxFrameIndex, nodeCount);
97    }
