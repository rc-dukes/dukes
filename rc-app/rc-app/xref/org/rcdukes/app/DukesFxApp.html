3   import java.net.URL;

5   import org.kohsuke.args4j.CmdLineException;
6   import org.kohsuke.args4j.CmdLineParser;
7   import org.kohsuke.args4j.Option;
8   import org.rcdukes.opencv.NativeLibrary;
9   
10  import javafx.application.Application;
11  import javafx.fxml.FXMLLoader;
12  import javafx.scene.Scene;
13  import javafx.scene.layout.VBox;
14  import javafx.stage.Stage;
15  
16  /**
17   * Java FX based GUI
18   * 
19   * @author wf
20   *
21   */
22  public class DukesFxApp extends Application {
23  
24    private static String[] cmdLineArgs;
25    private CmdLineParser parser;
26    @Option(name = "-d", aliases = {
27        "--debug" }, usage = "debug\ncreate additional debug output if this switch is used")
28    protected boolean debug = false;
29  
30    @Option(name = "-a", aliases = {
31        "--autostart" }, usage = "autostart using the configuration")
32    protected boolean autostart=false;
33    
34    @Option(name = "-iw", aliases = {
35    "--imageWidth" }, usage = "camera imageWidth")
36    protected int imageWidth=800;
37    
38    @Option(name = "-s", aliases = {
39    "--simulator" }, usage = "configure simulator")
40    protected boolean simulator=false;
41  
42    /**
43     * get the resource from the given path
44     * 
45     * @param path
46     *          - relative to src/main/resource in the target classpath
47     * @return - the URL for the resource
48     */
49    public URL getResource(String path) {
50      return getClass().getClassLoader().getResource(path);
51    }
52  
53    @Override
54    public void start(Stage primaryStage) throws Exception {
55      this.parseArguments();
56      // see
57      // https://stackoverflow.com/questions/39164050/javafx-8-tabpanes-and-tabs-with-separate-fxml-and-controllers-for-each-tab
58      FXMLLoader loader = new FXMLLoader(getResource("fx/dukes.fxml"));
59      VBox root = loader.load();
60      Scene scene = new Scene(root, BaseGUI.getScreenWidth() * 3 / 4,
61          BaseGUI.getScreenHeight() * 3 / 4);
62      String css = getResource("fx/dukes.css").toExternalForm();
63      scene.getStylesheets().add(css);
64      DukesFxGUI gui = loader.getController();
65      gui.init(primaryStage);
66      primaryStage.setTitle("RC-Dukes Self driving Car");
67      primaryStage.setScene(scene);
68      primaryStage.setMaximized(true);
69      primaryStage.widthProperty().addListener((obs, oldVal, newVal) -> {
70        gui.addJustFullScreenButton();
71      });
72      primaryStage.show();
73      gui.setDebug(debug);
74      gui.setCameraImageWidth(imageWidth);
75      if (autostart) {
76        gui.autoStart(simulator);
77      }
78    }
79  
80    /**
81     * parse the command line arguments
82     * 
83     * @param args
84     * @throws CmdLineException
85     */
86    public void parseArguments() throws CmdLineException {
87      parser = new CmdLineParser(this);
88      parser.parseArgument(cmdLineArgs);
89    }
90  
91    /**
92     * run the app
93     * 
94     * @param args
95     * @throws Exception
96     */
97    public static void main(String... args) throws Exception {
98      cmdLineArgs = args;
99      NativeLibrary.load();
100     launch(args);
101   }
102 }
