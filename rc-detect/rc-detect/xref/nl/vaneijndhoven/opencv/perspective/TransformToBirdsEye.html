1   package nl.vaneijndhoven.opencv.perspective;

3   import nl.vaneijndhoven.dukes.geometry.Polygon;
4   import nl.vaneijndhoven.opencv.mapper.PointMapper;
5   import org.opencv.core.Mat;
6   import org.opencv.core.MatOfPoint2f;
7   import org.opencv.core.Point;
8   import org.opencv.imgproc.Imgproc;
9   
10  import java.util.List;
11  import java.util.stream.Collectors;
12  
13  
14  public class TransformToBirdsEye {
15  
16      private final Polygon imagePolygon;
17      private final Polygon worldPolygon;
18      Mat perspectiveTransform;
19  
20      public TransformToBirdsEye(Polygon imagePolygon, Polygon worldPolygon) {
21          this.imagePolygon = imagePolygon;
22          this.worldPolygon = worldPolygon;
23          this.perspectiveTransform = buildPerspectiveTransform();
24      }
25  
26      private Mat buildPerspectiveTransform() {
27  
28          List<Point> imagePoints = this.imagePolygon.getPointsCounterClockwise().stream().map(PointMapper::toPoint).collect(Collectors.toList());
29          List<Point> worldPoints = this.worldPolygon.getPointsCounterClockwise().stream().map(PointMapper::toPoint).collect(Collectors.toList());
30  
31          MatOfPoint2f imagePointsMat = new MatOfPoint2f();
32          MatOfPoint2f worldPointsMat = new MatOfPoint2f();
33  
34          imagePointsMat.fromList(imagePoints);
35          worldPointsMat.fromList(worldPoints);
36  
37          return Imgproc.getPerspectiveTransform(imagePointsMat, worldPointsMat);
38      }
39  
40      public Mat transform(Mat input) {
41          Mat output = new Mat();
42          Imgproc.warpPerspective(input, output, perspectiveTransform, input.size());
43          return output;
44      }
45  }
