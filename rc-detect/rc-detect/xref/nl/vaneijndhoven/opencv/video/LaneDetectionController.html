125     private void verifyAngleFound(double angle, long currentTime) {
126         if (!(angle > 0) && !(angle < 0)) {
127             // no angle detected
128             if ((currentTime - tsLastLinesDetected > MAX_DURATION_NO_LINES_DETECTED) && !emergencyStopActivated) {
129                 System.out.println("no angle found for " + MAX_DURATION_NO_LINES_DETECTED + "ms, emergency stop");
130                 if (vertx != null) {
131                     emergencyStopActivated = true;
132                     vertx.eventBus().send("control", "speed:stop");
133                 }
134             }
135         } else {
136             // all good
137             tsLastLinesDetected = currentTime;
138         }
139     }

190     private void eventBusSendAfterMS(long ms, String command) {
191         new java.util.Timer().schedule(
192                 new java.util.TimerTask() {
193                     @Override
194                     public void run() {
195 //                        System.out.println("Sending command '" + command + "'.");
196                         if (vertx != null) {
197                             vertx.eventBus().send("control", command);
198                         } else {
199                             System.out.println("Couldn't send command '" + command + "', Vert.x not inited");
200                         }
201                     }
202                 },
203                 ms
204         );
205     }
