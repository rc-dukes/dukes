10  public class MemoryManagement {
11  
12      public static <S extends Mat> ClosableMat<S> closable(S resource) {
13          return new ClosableMat<>(resource);
14      }
15  
16      public static <T, S extends Mat> T scoped(Supplier<S> resource, Function<S, T> execution) {
17          try(ClosableMat<S> res = new ClosableMat<>(resource.get())) {
18              return execution.apply(res.get());
19          }
20      }
21  
22      public static <T, S extends Mat> T scoped(String title, Supplier<S> resource, Function<S, T> execution) {
23          return scoped(resource, execution);
24      }
25  
26      public static <S extends Mat> void scoped(Supplier<S> resource, Consumer<S> execution) {
27          try(ClosableMat<S> res = new ClosableMat<>(resource.get())) {
28              execution.accept(res.get());
29          }
30      }
31  
32      public static <S extends Mat> void scoped(String title, Supplier<S> resource, Consumer<S> execution) {
33          scoped(resource, execution);
34      }
35  
36      public static class ClosableMat<S extends Mat> extends Mat implements AutoCloseable {
37  
38          private S mat;
39  
40          public ClosableMat(S mat) {
41              this.mat = mat;
42          }
43  
44          @Override
45          public void close() {
46              mat.release();
47          }
48  
49          public S get() {
50              return mat;
51          }
52      }
53  
54      public static Observable<Mat> disposable(Mat mat) {
55          return Observable.using(() -> mat, Observable::just, Mat::release);
56      }
