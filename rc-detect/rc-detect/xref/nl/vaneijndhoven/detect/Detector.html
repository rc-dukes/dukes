55      LOG.info("Starting Daisy (image processing)");
56      vertx.eventBus().consumer(Events.STREAMADDED.name(), this::streamAdded);
57  
58      vertx.eventBus().consumer(
59          Characters.DAISY.getCallsign() + ":" + START_LANE_DETECTION,
60          this::startLD);
61      // vertx.eventBus().consumer(Characters.DAISY.getCallsign() + ":" +
62      // START_STARTLIGHT_DETECTION, this::startSLD);
63  
64      vertx.eventBus().consumer(
65          Characters.DAISY.getCallsign() + ":" + CANNY_CONFIG_UPDATE,
66          this::cannyConfig);
67      vertx.eventBus().consumer(
68          Characters.DAISY.getCallsign() + ":" + HOUGH_CONFIG_UPDATE,
69          this::houghConfig);
70      vertx.eventBus().consumer(
71          Characters.DAISY.getCallsign() + ":" + CAMERA_MATRIX_UPDATE,
72          this::cameraMatrix);
73  
74      LOG.info("Daisy started");

88          // .doOnNext(detection -> LOG.trace("Image lane detection processing

171   private Observable<String> startLaneDetection(Message<JsonObject> msg) {
172     JsonObject jo = msg.body();
173     JsonObject config = jo.getJsonObject("config");
174 
175     CannyEdgeDetector.Config canny = new CannyEdgeDetector.Config();
176     ProbabilisticHoughLinesLineDetector.Config hough = new ProbabilisticHoughLinesLineDetector.Config();
177     long interval = LANE_DETECTION_INTERVAL;
178     if (config != null) {
179       JsonObject cannyCfg = config.getJsonObject("canny");
180       if (cannyCfg != null) {
181         canny = new CannyEdgeDetector.Config();
182         canny.setThreshold1(cannyCfg.getDouble("threshold1"));
183         canny.setThreshold2(cannyCfg.getDouble("threshold2"));
184       }
185 
186       JsonObject houghCfg = config.getJsonObject("hough");
187       if (houghCfg != null) {
188         hough = new ProbabilisticHoughLinesLineDetector.Config();
189         hough.setRho(houghCfg.getDouble("rho"));
190         hough.setTheta(houghCfg.getDouble("theta"));
191         hough.setThreshold(houghCfg.getInteger("threshold"));
192         hough.setMaxLineGap(houghCfg.getDouble("maxLineGap"));
193         hough.setMinLineLength(houghCfg.getDouble("minLineLength"));
194       }
195 
196       if (config.containsKey("interval")) {
197         interval = config.getLong("interval");
198       }
199     }
200 
201     return startLaneDetection(jo.getString("source"), interval).map(map -> {
202       try {
203         return new ObjectMapper().writeValueAsString(map);
204       } catch (JsonProcessingException e) {
205         throw new RuntimeException(e);
206       }
207     });
208   }

210   private Observable<Object> startLaneDetection(String source, long interval) {
211     ImageFetcher fetcher = new ImageFetcher(source);
212 
213     LOG.info("Started image processing for source: " + source);
214     return fetcher.toObservable()
215         // .sample(interval, TimeUnit.MILLISECONDS)
216         .doOnNext(frame -> Detector.MAT = frame).map(frame -> {
217           ImageCollector collector = new ImageCollector();
218           Map<String, Object> detection = new LaneDetector(createCanny(),
219               createHoughLines(), createMatrix(), collector).detect(frame);
220           Detector.CANNY_IMG = collector.edges();
221           // detection.put("mat", frame.getNativeObjAddr());
222           return detection;
223         });
224   }

226   private Observable startStartLightDetection(Message<JsonObject> msg) {
227     JsonObject jo = msg.body();
228     JsonObject config = jo.getJsonObject("config");
229 
230     long interval = START_LIGHT_DETECTION_INTERVAL;
231     if (config != null) {
232       if (config.containsKey("interval")) {
233         interval = config.getLong("interval");
234       }
235     }
236 
237     StartLightDetector.Config config1 = new StartLightDetector.Config();
238     StartLightDetector startLightDetector = new StartLightDetector(config1);
239 
240     ImageFetcher fetcher = new ImageFetcher(jo.getString("source"));
241 
242     return fetcher.toObservable().sample(interval, TimeUnit.MILLISECONDS)
243         .map(startLightDetector::detect).map(map -> {
244           try {
245             return new ObjectMapper().writeValueAsString(map);
246           } catch (JsonProcessingException e) {
247             throw new RuntimeException(e);
248           }
249         });
250   }
