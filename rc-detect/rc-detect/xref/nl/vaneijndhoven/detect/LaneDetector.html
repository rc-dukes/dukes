13  /**
14   * detector for lanes
15   *
16   */
17  public class LaneDetector {
18  
19      public static double DEFAULT_LANE_BOUNDARY_ANGLE = 45;
20      public static double DEFAULT_LANE_BOUNDARY_TOLERANCE = 30;
21  
22      public static final double DEFAULT_CANNY_THRESHOLD_1 = 131d;
23      public static final double DEFAULT_CANNY_THRESHOLD_2 = 397d;
24      public static final int DEFAULT_LINE_DETECT_RHO = 1;
25      public static final double DEFAULT_LINE_DETECT_THETA = Math.PI / 180;
26      public static final double DEFAULT_LINE_DETECT_THRESHOLD = 42d;
27      public static final double DEFAULT_LINE_DETECT_MIN_LINE_LENGTH = 0d;
28      public static final double DEFAULT_LINE_DETECT_MAX_LINE_GAP = 98d;
29  
30      private CannyEdgeDetector.Config cannyConfig;
31      private ProbabilisticHoughLinesLineDetector.Config lineDetectorConfig;
32      private CameraMatrix matrix;
33      private ImageCollector collector;
34  
35  
36      public LaneDetector(CannyEdgeDetector.Config cannyConfig, ProbabilisticHoughLinesLineDetector.Config lineDetectorConfig, CameraMatrix matrix, ImageCollector collector) {
37          this.cannyConfig = cannyConfig;
38          this.lineDetectorConfig = lineDetectorConfig;
39          this.matrix = matrix;
40          this.collector = collector;
41      }
42  
43      public Map<String, Object> detect(Mat frame) {
44          return performLaneDetection(frame);
45  
46      }
47  
48      public Map<String, Object> performLaneDetection(Mat originalImage) {
49          ImageLaneDetection laneDetect = new ImageLaneDetection(cannyConfig, lineDetectorConfig, matrix);
50          return laneDetect.detectLane(originalImage, collector);
51      }
52  
53      public void setCannyConfig(CannyEdgeDetector.Config cannyConfig) {
54          this.cannyConfig = cannyConfig;
55      }
56  
57      public void setLineDetectorConfig(ProbabilisticHoughLinesLineDetector.Config lineDetectorConfig) {
58          this.lineDetectorConfig = lineDetectorConfig;
59      }
60  
61      public void setCollector(ImageCollector collector) {
62          this.collector = collector;
63      }
64  }
