53      super.preStart();
54      NativeLibrary.load();
55      // register vert.x event handlers
56      consumer(Events.START_LANE_DETECTION,
57          this::startLaneDetectionMessageHandler);
58      // consumer(Events.START_STARTLIGHT_DETECTION, this::startSLD); - blocking issue ..
59      consumer(Events.CAMERA_CONFIG_UPDATE, this::cameraConfig);
60      consumer(Events.CANNY_CONFIG_UPDATE, this::cannyConfig);

62      consumer(Events.CAMERA_MATRIX_UPDATE, this::cameraMatrix);
63      super.postStart();

66    /**
67     * start Lane Detection
68     * 
69     * @param message
70     */
71    private void startLaneDetectionMessageHandler(Message<JsonObject> message) {
72      JsonObject jo = message.body();
73      CameraConfig cameraConfig = super.fromJsonObject(jo, CameraConfig.class);
74      // is this a restart?
75      if (startLaneDetectionSubscription != null)
76        startLaneDetectionSubscription.dispose();;
77      if (fetcher!=null)
78        fetcher.close();
79      startLaneDetectionSubscription = startLaneDetection(cameraConfig)
80          // .doOnNext(detection -> LOG.trace("Image lane detection processing
81          // result: " + detection))
82          .subscribe(
83              lane -> {
84                JsonObject laneJo=JsonObject.mapFrom(lane);
85                super.sendEvent(Characters.LUKE, Events.LANE_DETECTED, laneJo);
86              },
87              error -> LOG.error("Error during lane detection image processing:",
88                  error),
89              () -> LOG.info("Lane detection image processing ended"));
90    }

92    private void startSLD(Message<JsonObject> message) {
93      // is this a restart?
94      if (startLightDetectionSubscription != null)
95        startLightDetectionSubscription.dispose();;
96      startLightDetectionSubscription = startStartLightDetection(message)
97          // .doOnNext(detection -> LOG.trace("Image start light processing
98          // result: " + detection))
99          // .takeWhile()
100         .subscribe(
101             light -> vertx.eventBus()
102                 .publish(Events.START_STARTLIGHT_DETECTION.name(), light),
103             error -> LOG.error("Error during start light image processing:",
104                 error),
105             () -> LOG.info("Start light image processing ended"));
106   }

108   private void cameraConfig(Message<JsonObject> message) {
109     JsonObject jo = message.body();
110     JsonObject currentJo = this.getSharedData("cameraConfig");
111     boolean restart = false;
112     if (currentJo == null) {
113       restart = true;
114     } else {
115       CameraConfig currentCameraConfig = currentJo.mapTo(CameraConfig.class);
116       CameraConfig cameraConfig = jo.mapTo(CameraConfig.class);
117       restart = !cameraConfig.getSource()
118           .equals(currentCameraConfig.getSource());
119     }
120     // update the camera Configuration
121     shareData("cameraConfig", jo);
122     if (restart) {
123       sendEvent(this.character,Events.START_LANE_DETECTION, jo);
124       sendEvent(this.character,Events.START_STARTLIGHT_DETECTION, jo);
125     }
126   }

160   /**
161    * startLaneDetectionObservable
162    * 
163    * @param cameraConfig
164    * @return
165    */
166   private Observable<LaneDetectionResult> startLaneDetectionObservable(
167       CameraConfig cameraConfig) {
168     fetcher = new ImageFetcher(cameraConfig.getSource());
169     fetcher.setFps(cameraConfig.getFps());
170 
171     LOG.info(
172         "Started image processing for source: " + cameraConfig.getSource());
173     return fetcher.toObservable().subscribeOn(Schedulers.newThread())
174         .throttleFirst(cameraConfig.getInterval(), TimeUnit.MILLISECONDS)
175         .map(image -> {
176           currentCollector = new ImageCollector();
177           EdgeDetector edgeDetector = super.getSharedPojo("canny",
178               CannyEdgeDetector.class);
179           LineDetector lineDetector = super.getSharedPojo("hough",
180               HoughLinesLineDetector.class);
181           CameraConfig updatedCameraConfig = super.getSharedPojo("cameraConfig",
182               CameraConfig.class);
183           LaneDetectionResult detection = new LaneDetector(edgeDetector,
184               lineDetector, updatedCameraConfig, createMatrix(),
185               currentCollector).detect(image);
186           return detection;
187         });
188   }

190   private Observable startStartLightDetection(Message<JsonObject> msg) {
191     JsonObject jo = msg.body();
192     JsonObject config = jo.getJsonObject("config");
193 
194     long interval = START_LIGHT_DETECTION_INTERVAL;
195     if (config != null) {
196       if (config.containsKey("interval")) {
197         interval = config.getLong("interval");
198       }
199     }
200 
201     StartLightDetectorImpl.Config config1 = new StartLightDetectorImpl.Config();
202     StartLightDetectorImpl startLightDetector = new StartLightDetectorImpl(
203         config1);
204 
205     // @FIXME - don't create multiple image fetchers
206     ImageFetcher fetcher = new ImageFetcher(jo.getString("source"));
207 
208     return fetcher.toObservable().sample(interval, TimeUnit.MILLISECONDS)
209         .map(startLightDetector::detect).map(map -> {
210           try {
211             return new ObjectMapper().writeValueAsString(map);
212           } catch (JsonProcessingException e) {
213             throw new RuntimeException(e);
214           }
215         });
216   }
