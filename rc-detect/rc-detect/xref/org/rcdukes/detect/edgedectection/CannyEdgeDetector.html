1   package org.rcdukes.detect.edgedectection;
2   
3   import org.opencv.core.Mat;
4   import org.opencv.imgproc.Imgproc;
5   import org.rcdukes.detectors.EdgeDetector;
6   
7   /**
8    * Canny edge detector
9    * @see <a href="https://docs.opencv.org/trunk/da/d22/tutorial_py_canny.html">OpenCV Canny Edge Detection </a>
10   */
11  public class CannyEdgeDetector implements EdgeDetector {
12  
13    double threshold1 = 60;
14    double threshold2 = 150;
15    int apertureSize = 3;
16    boolean l2gradient = false;
17    
18    public double getThreshold1() {
19      return threshold1;
20    }
21  
22    public void setThreshold1(double threshold1) {
23      this.threshold1 = threshold1;
24    }
25  
26    public double getThreshold2() {
27      return threshold2;
28    }
29  
30    public void setThreshold2(double threshold2) {
31      this.threshold2 = threshold2;
32    }
33  
34    public int getApertureSize() {
35      return apertureSize;
36    }
37  
38    public void setApertureSize(int apertureSize) {
39      this.apertureSize = apertureSize;
40    }
41  
42    public boolean isL2gradient() {
43      return l2gradient;
44    }
45  
46    public void setL2gradient(boolean l2gradient) {
47      this.l2gradient = l2gradient;
48    }
49  
50    public CannyEdgeDetector() {
51    }
52  
53    /**
54     * construct me with the given parameters
55     * @param threshold1
56     * @param threshold2
57     * @param apertureSize
58     * @param l2gradient
59     */
60    public CannyEdgeDetector(double threshold1, double threshold2,
61        int apertureSize, boolean l2gradient) {
62      this();
63      this.threshold1 = threshold1;
64      this.threshold2 = threshold2;
65      this.apertureSize = apertureSize;
66      this.l2gradient = l2gradient;
67    }
68  
69    @Override
70    public Mat detect(Mat image) {
71      Mat imgEdges = new Mat();
72      Imgproc.Canny(image, imgEdges, threshold1, threshold2, apertureSize,
73          l2gradient);
74      return imgEdges;
75    }
76  
77  }
