3   import nl.vaneijndhoven.opencv.tools.MemoryManagement;
4   import org.opencv.core.Mat;
5   import org.opencv.videoio.VideoCapture;
6   import rx.Observable;
7   import rx.functions.Action1;
8   import rx.functions.Func0;
9   import rx.functions.Func1;
10  
11  public class ImageFetcher {
12  
13      private VideoCapture capture = new VideoCapture();
14      private String source;
15  
16      public ImageFetcher(String source) {
17          this.source = source;
18      }
19  
20      public Mat fetch() {
21          if (!this.capture.isOpened()) {
22              this.capture.open(source);
23              throw new IllegalStateException("Trying to fetch image from unopened VideoCapture");
24          }
25  
26          final Mat frame = new Mat();
27  
28          this.capture.read(frame);
29  
30          return !frame.empty() ? frame : null;
31      }
32  
33      @Override
34      protected void finalize() throws Throwable {
35          super.finalize();
36      }
37  
38      public Observable<Mat> toObservable() {
39          // Resource creation.
40          Func0<VideoCapture> resourceFactory = () -> {
41              VideoCapture capture = new VideoCapture();
42              capture.open(source);
43              return capture;
44          };
45  
46          // Convert to observable.
47          Func1<VideoCapture, Observable<Mat>> observableFactory =
48                  capture -> Observable.<Mat>create(subscription -> {
49                      boolean hasNext = true;
50                      while (hasNext) {
51                          final Mat frame = new Mat();
52  
53                          hasNext = capture.read(frame);
54  
55                          subscription.onNext(frame);
56                      }
57  
58                      subscription.onCompleted();
59                  });//.switchMap(MemoryManagement::disposable);
60  
61          // Disposal function.
62          Action1<VideoCapture> dispose = VideoCapture::release;
63  
64          return Observable.using(resourceFactory, observableFactory, dispose);
65      }
66  }
