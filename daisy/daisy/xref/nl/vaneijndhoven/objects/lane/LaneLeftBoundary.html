12  public class LaneLeftBoundary extends Boundary {
13  
14      public static final double DEFAULT_ANGLE = -LaneDetectionController.DEFAULT_LANE_BOUNDARY_ANGLE;
15      public static final double DEFAULT_TOLERANCE = LaneDetectionController.DEFAULT_LANE_BOUNDARY_TOLERANCE;
16  
17      public LaneLeftBoundary() {
18          this(DEFAULT_ANGLE, DEFAULT_TOLERANCE, false);
19      }
20  
21      public LaneLeftBoundary(double angle, double margin, boolean directional) {
22          super(new LineFilter(angle, margin, directional));
23      }
24  
25      public Optional<Line> boundary(Collection<Line> lines) {
26          Comparator<Line> yComparator = (line1, line2) -> Double.compare(line1.bottomMost().getY(), line2.bottomMost().getY());
27          Comparator<Line> xComparator = (line1, line2) -> Double.compare(line1.bottomMost().getX(), line2.bottomMost().getX());
28          Comparator<Line> lineComparator = yComparator.thenComparing(xComparator);
29          Optional<Line> nearest = candidates(lines).stream().sorted(lineComparator).findFirst();
30          return nearest;
31  //        return Line.average(candidates(lines));
32      }
33  }
