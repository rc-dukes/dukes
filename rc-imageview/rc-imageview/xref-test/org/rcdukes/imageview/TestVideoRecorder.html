34    @Test
35    public void testVideoRecorder() throws IOException {
36      DebugImageServer.imageFormat=ImageFormat.jpg;
37      ImageCollector imageCollector=new ImageCollector();
38      Image testImage=imageCollector.getImage(ImageType.camera,true);
39      Mat testMat = testImage.getFrame();
40  
41      for (String ext : VideoRecorder.exts) {
42        for (String FOURCC : VideoRecorder.FOURCCs) {
43          FOURCC = FOURCC.toLowerCase();
44          VideoRecorder recorder = new VideoRecorder("test", 25);
45          recorder.setExt(ext);
46          recorder.FOURCC = FOURCC;
47          String msg = String.format(
48              "recording: %dx%d %d channel %s video with %s", testMat.width(),
49              testMat.height(), testMat.channels(), ext, FOURCC);
50          System.out.println(msg);
51          for (int i = 1; i <= 50; i++) {
52            recorder.recordMat(testMat);
53          }
54          recorder.stop();
55          File videoFile = new File(recorder.getPath());
56          assertTrue(videoFile.exists());
57          assertTrue(videoFile.length() > 50000);
58        }
59      }
60    }

62    @Test
63    public void testVideoRecorders() throws IOException {
64      VideoRecorders vr=new VideoRecorders(10.);
65      ImageCollector imageCollector=new ImageCollector();
66      String imagePath="images/road.jpg";
67      vr.start();
68      Mat frame = ImageUtils.fromResource(this.getClass(), imagePath);
69      imageCollector.addImage(frame, ImageType.camera);
70      vr.recordFrame(imageCollector,true);
71      VideoInfo info = vr.stop();
72      if (debug) {
73        JsonObject infoJo = JsonObject.mapFrom(info);
74        System.out.println(infoJo.encodePrettily());
75      }
76      assertEquals(10.0,info.fps,0.01);
77      assertEquals(1,info.minFrameIndex.intValue());
78      assertEquals(1,info.maxFrameIndex.intValue());
79      assertFalse(info.path.contains(".avi"));
80      assertTrue(info.path.endsWith(".json"));
81      assertTrue(info.path.contains("navigation_")); 
82    }

84    @Test
85    public void testVideoInfoGraphDatabase() throws IOException {
86      TinkerPopDatabase tpd = new TinkerPopDatabase();
87      VideoInfo vInfo=new VideoInfo(25.);
88      vInfo.setFrameIndex(10);
89      vInfo.setFrameIndex(20);
90      tpd.addVertex(vInfo);
91      File gFile = File.createTempFile("videoInfo", ".json");
92      tpd.writeGraph(gFile.getPath());
93      TinkerPopDatabase tpd2=new TinkerPopDatabase();
94      tpd2.loadGraph(gFile);
95      assertEquals(1,tpd2.g().V().count().next().longValue());
96      Vertex vInfoNode=tpd2.g().V().hasLabel("VideoInfo").next();
97      VideoInfo vInfo2=tpd2.fromVertex(vInfoNode, VideoInfo.class);
98      assertEquals(25.,vInfo2.fps,0.01);
99      assertEquals(10,vInfo2.minFrameIndex.intValue());
100     assertEquals(20,vInfo2.maxFrameIndex.intValue());
101     gFile.delete();
102   }
