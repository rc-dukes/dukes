31    public void startImageServer() throws Exception {

33      clusterStarter = new ClusterStarter();
34      imageServer = new DebugImageServer();
35      clusterStarter.deployVerticles(imageServer);
36      imageServer.waitStatus(Status.started, TIME_OUT, 10);
37      assertNotNull(imageServer.deploymentID());
38    }

47    @Test
48    public void testTestImage() throws Exception {
49      startImageServer();
50      DebugImageServer.SERVE_TEST_IMAGES = true;
51      // @TODO Make configurable
52      String url = "http://localhost:8081";
53      AsyncHttpClient asyncHttpClient = asyncHttpClient();
54      Future<Response> whenResponse = asyncHttpClient.prepareGet(url).execute();
55      Response response = whenResponse.get();
56      assertEquals(200, response.getStatusCode());
57      byte[] imageBytes = response.getResponseBodyAsBytes();
58      Image image = ImageUtils.imageBytes2Image(imageBytes);
59      assertNotNull(image);
60      assertEquals(640, Math.round(image.getWidth()));
61      assertEquals(480, Math.round(image.getHeight()));
62    }
