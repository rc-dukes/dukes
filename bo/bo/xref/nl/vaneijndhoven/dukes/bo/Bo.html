1   package nl.vaneijndhoven.dukes.bo;
2   
3   import io.vertx.core.json.JsonObject;
4   import io.vertx.core.logging.Logger;
5   import io.vertx.core.logging.LoggerFactory;
6   import io.vertx.rxjava.core.AbstractVerticle;
7   import io.vertx.rxjava.core.eventbus.Message;
8   import nl.vaneijndhoven.dukes.car.Engine;
9   import nl.vaneijndhoven.dukes.car.Steering;
10  import nl.vaneijndhoven.dukes.car.Car;
11  import nl.vaneijndhoven.dukes.generallee.EngineMap;
12  import nl.vaneijndhoven.dukes.generallee.SteeringMap;
13  import nl.vaneijndhoven.dukes.hazardcounty.Characters;
14  import rx.Subscription;
15  
16  public class Bo extends AbstractVerticle {
17  
18      private static final Logger LOG = LoggerFactory.getLogger(Bo.class);
19  
20      private Car car = new Car(new Engine(new EngineMap()), new Steering(new SteeringMap()));
21      private SpeedHandler speedHandler = new SpeedHandler(car);
22      private SteeringHandler steeringHandler = new SteeringHandler(car);
23  
24      @Override
25      public void start() {
26          LOG.info("Starting Bo (motor and steering output)");
27  
28          Subscription subscription = vertx.eventBus().consumer(Characters.BO.getCallsign()).toObservable()
29                  .doOnNext(x -> LOG.trace("Received instruction"))
30                  .map(Message::body)
31                  .cast(JsonObject.class)
32                  .subscribe(message -> {
33                      LOG.trace("Instruction: {}", message);
34                      String type = message.getString("type");
35                      switch (type) {
36                          case "motor":
37                              speedHandler.handleMotor(message);
38                              break;
39                          case "servo":
40                              steeringHandler.handleServo(message);
41                              break;
42                          case "servoDirect":
43                              steeringHandler.handleServoDirect(message);
44                              break;
45                          case "speedDirect":
46                              speedHandler.handleSpeedDirect(message);
47                              break;
48                          case "log":
49                              String logMessage = message.getString("message");
50                              LOG.debug("Received log message: "  + logMessage);
51                              break;
52                          default:
53                              LOG.error("Unknown message type {}", type);
54                      }
55          });
56  
57          LOG.info("Bo started");
58      }
59  }
