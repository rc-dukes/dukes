1   package nl.vaneijndhoven.dukes.bo;
2   
3   import io.vertx.core.json.JsonObject;
4   import io.vertx.core.logging.Logger;
5   import io.vertx.core.logging.LoggerFactory;
6   import nl.vaneijndhoven.dukes.car.SteeringMap;
7   import nl.vaneijndhoven.dukes.car.Car;
8   
9   class SteeringHandler {
10  
11      private static final Logger LOG = LoggerFactory.getLogger(SteeringHandler.class);
12  
13      private static int currentWheelPosition;
14      private Car car;
15  
16      SteeringHandler(Car car) {
17          this.car = car;
18          this.currentWheelPosition = car.getSteering().getSteeringMap().center();
19      }
20  
21      void handleServoDirect(JsonObject messageBody) {
22          LOG.debug("Received direct message for servo: " + messageBody);
23          String position = messageBody.getString("position");
24  
25          int positionPercentage = Double.valueOf(position).intValue();
26  
27          SteeringMap steeringMap = car.getSteering().getSteeringMap();
28  
29          if (positionPercentage < 5) {
30              int range = steeringMap.center() - steeringMap.maxLeft();
31              currentWheelPosition = steeringMap.center() - ((range * Math.abs(positionPercentage))/100);
32              LOG.debug("directPos = " + position + ", range = " + range + ", newPos = " + currentWheelPosition);
33          } else if (positionPercentage > 5) {
34              int range = steeringMap.maxRight() - steeringMap.center();
35              currentWheelPosition = steeringMap.center() + ((range * Math.abs(positionPercentage))/100);
36              LOG.debug("directPos = " + position + ", range = " + range + ", newPos = " + currentWheelPosition);
37          } else {
38              currentWheelPosition = steeringMap.center();
39          }
40  
41          if (currentWheelPosition < steeringMap.maxLeft()) {
42              currentWheelPosition = steeringMap.maxLeft();
43          }
44          if (currentWheelPosition > steeringMap.maxRight()) {
45              currentWheelPosition = steeringMap.maxRight();
46          }
47          LOG.debug("about to set current wheel pos to " + currentWheelPosition);
48  //        Command.setWheelPosition(currentWheelPosition);
49          car.turn(currentWheelPosition);
50      }
51  
52      void handleServo(JsonObject messageBody) {
53          LOG.debug("Received message for servo: " + messageBody);
54          String position = messageBody.getString("position");
55  
56          SteeringMap steeringMap = car.getSteering().getSteeringMap();
57  
58          if ("left".equals(position)) {
59              currentWheelPosition = currentWheelPosition - steeringMap.stepSize();
60              if (currentWheelPosition < steeringMap.maxLeft()) {
61                  currentWheelPosition = steeringMap.maxLeft();
62              }
63          } else if ("right".equals(position)) {
64              currentWheelPosition = currentWheelPosition + steeringMap.stepSize();
65              if (currentWheelPosition > steeringMap.maxRight()) {
66                  currentWheelPosition = steeringMap.maxRight();
67              }
68          } else if ("center".equals(position)) {
69              currentWheelPosition = steeringMap.center();
70          }
71  //        Command.setWheelPosition(currentWheelPosition);
72          car.turn(currentWheelPosition);
73      }
74  
75  }
