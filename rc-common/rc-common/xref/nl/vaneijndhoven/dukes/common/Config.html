4   import ch.qos.logback.classic.Level;
5   import ch.qos.logback.classic.LoggerContext;
6   import io.vertx.core.logging.SLF4JLogDelegateFactory;
7   import io.vertx.core.spi.cluster.ClusterManager;
8   import io.vertx.spi.cluster.hazelcast.HazelcastClusterManager;
9   import org.slf4j.LoggerFactory;
10  
11  /**
12   * Configuration for Logging and HazelCastClusterManager
13   *
14   */
15  public class Config {
16  
17      /**
18       * configure the logging
19       */
20      public static void configureLogging() {
21          System.setProperty(io.vertx.core.logging.LoggerFactory.LOGGER_DELEGATE_FACTORY_CLASS_NAME, SLF4JLogDelegateFactory.class.getName());
22  
23          LoggerContext logConfig = (LoggerContext) LoggerFactory.getILoggerFactory();
24          logConfig.getLogger("ROOT").setLevel(Level.TRACE);
25  
26          logConfig.getLogger("io.vertx").setLevel(Level.INFO);
27          logConfig.getLogger("com.hazelcast").setLevel(Level.INFO);
28          logConfig.getLogger("com.hazelcast.nio.tcp").setLevel(Level.ERROR);
29          logConfig.getLogger("io.netty").setLevel(Level.INFO);
30      }
31  
32      /**
33       * create a ClusterManager with a default hazelcast configuration using slf4j
34       * @return - the ClusterManager
35       */
36      public static ClusterManager createHazelcastConfig() {
37          com.hazelcast.config.Config hazelcastConfig = new com.hazelcast.config.Config();
38          hazelcastConfig.setProperty("hazelcast.logging.type", "slf4j");
39          ClusterManager mgr = new HazelcastClusterManager(hazelcastConfig);
40          return mgr;
41      }
42  
43  }
