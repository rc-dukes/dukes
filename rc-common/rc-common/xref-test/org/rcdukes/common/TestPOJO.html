12  /**
13   * test Plain Old Java Object conversion with vert.x
14   * @author wf
15   *
16   */
17  public class TestPOJO {
18    boolean debug=true;
19    @Test
20    /**
21     * test Vert.x pojo support
22     * see <a href='http://vertx.io/docs/apidocs/io/vertx/core/json/JsonObject.html>JsonObject</a>;
23     */
24    public void testPOJO() {
25      // https://stackoverflow.com/a/42003953/1497139
26      ServoCalibration sc = new ServoCalibration();
27      sc.setVehicle("Ford");
28      sc.setType("engine");
29      sc.setUnit("m/s");
30      sc.getValueMap().put(141, 1.19);
31      sc.getValueMap().put(142, 1.24);
32      sc.getValueMap().put(143, 1.47);
33      sc.getValueMap().put(144, 1.8);
34      sc.getValueMap().put(145, 2.0);
35      sc.getValueMap().put(146, 2.2);
36      sc.getValueMap().put(147, 2.2);    
37      
38      Gson gson = new Gson();
39      JsonObject jo = sc.asJsonObject();
40      String vjson=jo.toString();
41      String gjson=gson.toJson(sc);
42      if (debug) {
43        System.out.println(vjson);
44        System.out.println(sc);
45      }
46      assertEquals(vjson,gjson);
47      ServoCalibration sc2 = ServoCalibration.fromJo(jo);
48      assertEquals(sc.asJson(),sc2.asJson());
49    }
50  }
