19  public class CameraMatrix implements UnaryOperator<Mat> {
20  
21      public static final CameraMatrix DEFAULT = new CameraMatrix(0, 0);
22      Mat cameraMatrix = null;
23      Mat distCoeffs = null;
24  
25      private List<Mat> rvecs = null;
26      private List<Mat> tvecs = null;
27  
28      List<Mat> imagePoints = new ArrayList<>();
29      List<Mat> objectPoints = new ArrayList<>();
30  
31      private int columns;
32      private int rows;
33  
34      public CameraMatrix(int columns, int rows) {
35          this.columns = columns;
36          this.rows = rows;
37      }
38  
39      public void calibrate(Mat... images) {
40          for (Mat image : images) {
41              addCalibrationImage(image);
42          }
43      }
44  
45      private void calculate(Size size) {
46          if (imagePoints.isEmpty()) {
47              System.err.println("Could not find chessboard corners");
48              return;
49          }
50  
51          Mat cameraMatrix = new Mat();
52          Mat distCoeffs = new Mat();
53          List<Mat> rvecs = new ArrayList<>();
54          List<Mat> tvecs= new ArrayList<>();
55  
56          Calib3d.calibrateCamera(objectPoints, imagePoints, size, cameraMatrix, distCoeffs, rvecs, tvecs);
57  
58          this.cameraMatrix = cameraMatrix;
59          this.distCoeffs = distCoeffs;
60          this.rvecs = rvecs;
61          this.tvecs = tvecs;
62      }
63  
64      private void addCalibrationImage(Mat image) {
65          Mat grey = new Mat();
66          Imgproc.cvtColor(image, grey, COLOR_BGR2GRAY);
67  
68          columns = 8;
69          rows = 6;
70          MatOfPoint2f corners = new MatOfPoint2f();
71          boolean patternWasFound = findCorners(grey, columns, rows, corners);
72  
73          Calib3d.drawChessboardCorners(image, new Size(columns, rows), corners, patternWasFound);
74  
75          if (!patternWasFound) {
76              return;
77          }
78  
79          imagePoints.add(corners);
80  
81          MatOfPoint3f obj = new MatOfPoint3f();
82          for (int y = 0; y < rows; y++) {
83              for (int x = 0; x < columns; x++) {
84                  obj.push_back(new MatOfPoint3f(new Point3(y, x, 0.0)));
85              }
86          }
87  
88          objectPoints.add(obj);
89  
90          calculate(image.size());
91      }
92  
93  
94      private boolean findCorners(Mat image, int columns, int rows, MatOfPoint2f corners) {
95          return Calib3d.findChessboardCorners(image, new Size(columns, rows), corners);
96      }

99      @Override
100     public Mat apply(Mat image) {
101 //        Objects.requireNonNull(cameraMatrix, "CameraMatrix must first be calibrated by calling calibrate");
102 
103         if (cameraMatrix == null) {
104             return image;
105         }
106 
107         Mat newImage = new Mat();
108         undistort(image, newImage, cameraMatrix, distCoeffs);
109         return newImage;
110     }

112     public String serialize() {
113         JsonObject matrix = serializeMat(cameraMatrix);
114         JsonObject dists = serializeMat(distCoeffs);
115 
116         return new JsonObject()
117                 .put("cameraMatrix", matrix)
118                 .put("distCoeffs", dists)
119                 .toString();
120     }
121 
122     private JsonObject serializeMat(Mat mat) {
123         JsonObject matrix = new JsonObject();
124         matrix.put("size", new JsonObject()
125                 .put("cols", mat.size().width)
126                 .put("rows", mat.size().height));
127 
128         matrix.put("type", mat.type());
129 
130         JsonArray entries = new JsonArray();
131 
132         for (int x = 0; x < mat.size().width; x++) {
133             for (int y = 0; y < mat.size().height; y++) {
134                 double[] vals = mat.get(y,x);
135                 JsonArray jsonVals = new JsonArray();
136 
137                 Arrays.stream(vals).forEach(jsonVals::add);
138 
139                 entries.add(new JsonObject()
140                         .put("x", x)
141                         .put("y", y)
142                         .put("val", jsonVals));
143             }
144         }
145 
146         matrix = matrix.put("entries", entries);
147         return matrix;
148     }
149 
150     public static CameraMatrix deserizalize(String serialized) {
151         JsonObject object = new JsonObject(serialized);
152 
153         JsonObject cameraMatrixJson = object.getJsonObject("cameraMatrix");
154         JsonObject distCoeffsJson = object.getJsonObject("distCoeffs");
155 
156         CameraMatrix cameraMatrix = new CameraMatrix(0, 0);
157 
158         cameraMatrix.cameraMatrix = deserializeMat(cameraMatrixJson);
159         cameraMatrix.distCoeffs = deserializeMat(distCoeffsJson);
160 
161         return cameraMatrix;
162     }
163 
164     private static Mat deserializeMat(JsonObject matJson) {
165         Mat mat = new Mat(
166                 matJson.getJsonObject("size").getInteger("rows"),
167                 matJson.getJsonObject("size").getInteger("cols"),
168                 matJson.getInteger("type"));
169 
170         matJson.getJsonArray("entries").forEach(obj -> {
171             JsonObject entry = (JsonObject)obj;
172 
173             double[] values = entry.getJsonArray("val").stream()
174                     .mapToDouble(x -> (Double)x)
175                     .toArray();
176 
177             mat.put(entry.getInteger("y"), entry.getInteger("x"), values);
178         });
179 
180         return mat;
181     }
